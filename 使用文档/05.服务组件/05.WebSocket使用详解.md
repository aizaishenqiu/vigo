# WebSocket 实时通讯使用详解

Vigo 框架提供完整的 WebSocket 实时通讯支持，包括连接管理、频道订阅、指令路由、心跳保活等功能。

---

## 一、快速开始

### 1.1 启用 WebSocket

在 `config.yaml` 中配置 WebSocket：

```yaml
websocket:
  enabled: true
  port: 9001
  max_connections: 10000
  heartbeat_interval: 30  # 心跳间隔（秒）
  message_timeout: 10     # 消息超时（秒）
```

### 1.2 基础使用

```go
// app/controller/ws.go
package controller

import (
    "vigo/framework/mvc"
    "vigo/framework/websocket"
)

// WebSocketHandler 处理 WebSocket 连接
func WebSocketHandler(c *mvc.Context) {
    // 升级到 WebSocket 连接
    conn, err := websocket.Upgrade(c)
    if err != nil {
        return
    }
    
    // 获取连接 ID
    connID := conn.ID()
    
    // 发送欢迎消息
    conn.Send(map[string]interface{}{
        "type": "welcome",
        "data": "欢迎连接 WebSocket 服务",
    })
    
    // 连接会自动被框架管理，无需手动维护
}
```

### 1.3 路由配置

```go
// app/route/route.go
func Init() {
    r := mvc.NewRouter()
    
    // WebSocket 路由
    r.GET("/ws", controller.WebSocketHandler)
    r.GET("/ws/:channel", controller.WebSocketHandler)
}
```

## 二、频道管理

### 2.1 加入频道

```go
// 客户端发送加入频道消息
{
    "type": "join",
    "channel": "chat_room_1"
}

// 服务端处理
func OnJoin(conn *websocket.Conn, message websocket.Message) {
    channel := message.Data["channel"].(string)
    
    // 加入频道
    err := websocket.JoinChannel(conn, channel)
    if err != nil {
        conn.Send(map[string]interface{}{
            "type":  "error",
            "error": "加入频道失败",
        })
        return
    }
    
    // 发送成功通知
    conn.Send(map[string]interface{}{
        "type":    "system",
        "message": fmt.Sprintf("已加入频道：%s", channel),
    })
}
```

### 2.2 频道广播

```go
// 向频道内所有用户广播消息
func BroadcastToChannel(channel string, data interface{}) {
    websocket.Broadcast(channel, map[string]interface{}{
        "type": "message",
        "data": data,
    })
}

// 示例：发送聊天消息
func SendChatMessage(roomID string, userID int, content string) {
    BroadcastToChannel("chat_"+roomID, map[string]interface{}{
        "user_id":  userID,
        "content":  content,
        "time":     helper.Time(),
    })
}
```

### 2.3 获取频道信息

```go
// 获取频道在线人数
count := websocket.GetChannelCount("chat_room_1")

// 获取所有频道
channels := websocket.GetAllChannels()

// 判断连接是否在频道中
inChannel := websocket.InChannel(conn, "chat_room_1")
```

## 三、指令系统

### 3.1 定义指令处理器

```go
// app/websocket/handlers.go
package websocket

import (
    "vigo/framework/websocket"
)

// InitHandlers 初始化指令处理器
func InitHandlers() {
    // 注册指令处理器
    websocket.RegisterHandler("chat", HandleChat)
    websocket.RegisterHandler("gift", HandleGift)
    websocket.RegisterHandler("like", HandleLike)
    websocket.RegisterHandler("join", HandleJoin)
    websocket.RegisterHandler("leave", HandleLeave)
}

// HandleChat 处理聊天消息
func HandleChat(conn *websocket.Conn, message websocket.Message) {
    content := message.Data["content"].(string)
    
    // 保存消息到数据库
    // ...
    
    // 广播给频道内所有人
    websocket.Broadcast(message.Channel, map[string]interface{}{
        "type":    "chat",
        "user_id": conn.Get("user_id"),
        "content": content,
        "time":    helper.Time(),
    })
}

// HandleGift 处理送礼消息
func HandleGift(conn *websocket.Conn, message websocket.Message) {
    giftID := message.Data["gift_id"].(int)
    receiverID := message.Data["receiver_id"].(int)
    
    // 处理礼物逻辑
    // ...
    
    // 广播礼物消息
    websocket.Broadcast(message.Channel, map[string]interface{}{
        "type":       "gift",
        "sender_id":  conn.Get("user_id"),
        "receiver_id": receiverID,
        "gift_id":    giftID,
        "time":       helper.Time(),
    })
}
```

## 四、连接管理

### 4.1 连接认证

```go
// app/middleware/ws_auth.go
package middleware

import (
    "vigo/framework/mvc"
    "vigo/framework/websocket"
)

// WSAuth WebSocket 连接认证中间件
func WSAuth() mvc.Middleware {
    return func(c *mvc.Context) {
        // 获取 Token
        token := c.Query("token", "")
        
        // 验证 Token
        userID, err := verifyToken(token)
        if err != nil {
            c.AbortWithStatus(401)
            return
        }
        
        // 将用户信息存入上下文
        c.Set("user_id", userID)
        c.Next()
    }
}

// 在路由中使用
r.GET("/ws", controller.WebSocketHandler, middleware.WSAuth())
```

### 4.2 连接事件

```go
// app/websocket/events.go
package websocket

import (
    "vigo/framework/websocket"
)

// InitEvents 初始化连接事件
func InitEvents() {
    // 连接建立事件
    websocket.OnConnect(func(conn *websocket.Conn) {
        log.Printf("用户 %v 连接 WebSocket\n", conn.Get("user_id"))
        
        // 可以在这里做初始化操作
        // 如：加载用户信息、加入默认频道等
    })
    
    // 连接断开事件
    websocket.OnDisconnect(func(conn *websocket.Conn) {
        log.Printf("用户 %v 断开 WebSocket 连接\n", conn.Get("user_id"))
        
        // 清理操作
        // 如：退出所有频道、更新在线状态等
    })
    
    // 错误处理
    websocket.OnError(func(conn *websocket.Conn, err error) {
        log.Printf("WebSocket 错误：%v\n", err)
    })
}
```

## 五、完整示例

### 5.1 聊天室服务端

```go
// app/controller/chat.go
package controller

import (
    "vigo/framework/mvc"
    "vigo/framework/websocket"
    "vigo/framework/helper"
)

// ChatWebSocket 聊天室 WebSocket 处理
func ChatWebSocket(c *mvc.Context) {
    // 升级到 WebSocket
    conn, err := websocket.Upgrade(c)
    if err != nil {
        return
    }
    
    // 设置用户信息
    userID := c.GetInt64("user_id")
    conn.Set("user_id", userID)
    
    // 发送欢迎消息
    conn.Send(map[string]interface{}{
        "type": "welcome",
        "data": map[string]interface{}{
            "message":   "欢迎加入聊天室",
            "user_id":   userID,
            "time":      helper.Time(),
            "online":    websocket.GetTotalConnections(),
        },
    })
}

// HandleChatMessage 处理聊天消息
func HandleChatMessage(conn *websocket.Conn, message websocket.Message) {
    content := message.Data["content"].(string)
    if content == "" {
        conn.Send(map[string]interface{}{
            "type":  "error",
            "error": "消息内容不能为空",
        })
        return
    }
    
    // 构建消息
    chatMessage := map[string]interface{}{
        "type":     "chat",
        "from_id":  conn.Get("user_id"),
        "content":  content,
        "time":     helper.Datetime(),
        "timestamp": helper.Time(),
    }
    
    // 广播到频道
    if message.Channel != "" {
        websocket.Broadcast(message.Channel, chatMessage)
    } else {
        conn.Send(chatMessage)
    }
}
```

### 5.2 聊天室客户端

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>聊天室</title>
</head>
<body>
    <div id="messages"></div>
    <input type="text" id="messageInput" placeholder="输入消息...">
    <button onclick="sendMessage()">发送</button>

    <script>
        // 连接 WebSocket
        const ws = new WebSocket('ws://localhost:9001/ws?token=your_token');
        
        // 接收消息
        ws.onmessage = function(event) {
            const message = JSON.parse(event.data);
            addMessage(message);
        };
        
        // 发送消息
        function sendMessage() {
            const input = document.getElementById('messageInput');
            const content = input.value.trim();
            
            if (!content) return;
            
            ws.send(JSON.stringify({
                type: 'chat',
                channel: 'room_1',
                data: {
                    content: content
                }
            }));
            
            input.value = '';
        }
        
        // 添加消息到页面
        function addMessage(message) {
            const div = document.createElement('div');
            div.textContent = `[${message.from_id}]: ${message.content}`;
            document.getElementById('messages').appendChild(div);
        }
    </script>
</body>
</html>
```

## 六、高级功能

### 6.1 一对一私聊

```go
// 向指定用户发送消息
func SendPrivateMessage(fromID, toID int, content string) {
    // 获取目标用户的连接
    conn := websocket.GetConnectionByUserID(toID)
    if conn == nil {
        // 用户不在线
        return
    }
    
    conn.Send(map[string]interface{}{
        "type":      "private_message",
        "from_id":   fromID,
        "content":   content,
        "time":      helper.Time(),
    })
}
```

### 6.2 群组广播

```go
// 向多个用户广播
func BroadcastToUsers(userIDs []int, data interface{}) {
    for _, userID := range userIDs {
        conn := websocket.GetConnectionByUserID(userID)
        if conn != nil {
            conn.Send(data)
        }
    }
}
```

## 七、最佳实践

### 7.1 性能优化

- 合理设置心跳间隔（建议 30-60 秒）
- 控制最大连接数
- 及时清理断开的连接
- 使用连接池复用连接

### 7.2 安全建议

- 实施连接认证
- 限制单用户连接数
- 验证消息内容
- 防止消息 flooding

### 7.3 消息设计

- 统一消息格式
- 添加消息类型标识
- 包含时间戳
- 支持消息确认机制

通过 WebSocket，Vigo 框架可以构建实时聊天、在线通知、实时数据推送等各种实时应用。
