# 安全防护详细操作指南

本指南详细介绍了Vigo框架的安全防护机制和各种安全相关的操作方法。

## 1. XSS (跨站脚本) 防护

### 1.1 自动XSS过滤

框架内置了XSS防护中间件，自动对输入数据进行HTML转义：

```go
// 在config.yaml中启用XSS防护
security:
  xss:
    enable: true  # 开启XSS防护
    auto_escape: true  # 自动转义HTML特殊字符
```

### 1.2 手动XSS过滤

```go
import "vigo/framework/middleware"

// 手动过滤用户输入
func sanitizeInput(input string) string {
    return middleware.XSSFilter(input)
}

// 在控制器中使用
func (c *UserController) Store() {
    userInput := c.PostForm("comment")
    safeInput := sanitizeInput(userInput)
    // 使用安全的输入数据
}
```

### 1.3 HTML内容安全输出

```go
// 对于需要保留HTML格式的内容，使用白名单过滤
import "github.com/microcosm-cc/bluemonday"

func sanitizeHTMLContent(content string) string {
    p := bluemonday.UGCPolicy()
    // 允许基本的文本格式化标签
    p.AllowElements("p", "br", "strong", "em", "u")
    p.AllowAttrs("class").OnElements("p", "span")
    return p.Sanitize(content)
}
```

## 2. SQL注入防护

### 2.1 使用参数化查询

```go
// ✅ 正确：使用参数化查询（推荐）
userId := 123
user, err := db.Table("users").Where("id", userId).Find()

// ✅ 正确：使用占位符
name := "John"
users, err := db.Table("users").WhereRaw("name = ?", name).Select()

// ✅ 正确：多个参数
users, err := db.Table("users").WhereRaw("age > ? AND city = ?", 18, "Beijing").Select()

// ❌ 错误：直接拼接SQL（危险！）
// dangerousQuery := fmt.Sprintf("SELECT * FROM users WHERE id = %d", userId)
```

### 2.2 输入验证和清理

```go
// 创建输入验证器
type UserValidator struct {
    validator.BaseValidator
}

func (v *UserValidator) Rules() map[string]string {
    return map[string]string{
        "id":     "required|integer|min:1|max:999999",
        "name":   "required|string|max:50",
        "email":  "required|email|max:100",
        "age":    "required|integer|min:0|max:150",

    }
    
    // 验证通过后处理数据
    id := c.PostForm("id")
    name := c.PostForm("name")
    
    // 使用验证后的数据
    affected, err := db.Table("users").
        Where("id", id).
        Update(map[string]interface{}{"name": name})
}
```

## 3. CSRF (跨站请求伪造) 防护

### 3.1 启用CSRF中间件

```go
// 在路由中使用CSRF保护
func Init(r *mvc.Router) {
    // 全局启用CSRF保护（对于POST/PUT/DELETE请求）
    r.POST("/api/users", middleware.CSRF(), UserCreateHandler)
    r.PUT("/api/users/:id", middleware.CSRF(), UserUpdateHandler)
    r.DELETE("/api/users/:id", middleware.CSRF(), UserDeleteHandler)
}

// 在模板中生成CSRF令牌
func (c *HomeController) Index() {
    csrfToken := middleware.GenerateCSRFToken()
    c.Set("csrf_token", csrfToken)
    c.View("index.html")
}
```

### 3.2 模板中的CSRF令牌使用

```html
<!-- 在HTML表单中使用CSRF令牌 -->
<form method="POST" action="/api/users">
    <input type="hidden" name="_token" value="{{.csrf_token}}">
    <input type="text" name="name" placeholder="姓名">
    <input type="email" name="email" placeholder="邮箱">
    <button type="submit">提交</button>
</form>

<!-- 或者在AJAX请求中使用 -->
<script>
fetch('/api/users', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'X-CSRF-TOKEN': '{{.csrf_token}}'  // 通过HTTP头部传递
    },
    body: JSON.stringify({
        name: 'John',
        email: 'john@example.com'
    })
})
</script>
```

## 4. 速率限制 (DoS/DDoS 防护)

### 4.1 全局速率限制配置

```yaml
# config.yaml
security:
  dos:
    enable: true  # 是否开启DoS防护
    limit: 100    # 限制窗口内的最大请求数
    window: 60    # 限制窗口时长（秒）
    black_ip:     # 初始黑名单IP列表
      - 192.168.1.100
      - 10.0.0.50
```

### 4.2 自定义速率限制

```go
// 为特定路由设置不同的速率限制
func RateLimitMiddleware(maxRequests int, windowSeconds int) mvc.HandlerFunc {
    return func(c *mvc.Context) {
        ip := c.ClientIP()
        
        // 获取当前IP的请求次数
        key := fmt.Sprintf("rate_limit:%s", ip)
        current, err := redis.Incr(key)
        if err != nil {
            c.Error(429, "Rate limit check failed")
            c.Abort()
            return
        }
        
        if current == 1 {
            // 设置过期时间
            redis.Expire(key, windowSeconds)
        }
        
        if current > int64(maxRequests) {
            c.Error(429, "Too Many Requests")
            c.Abort()
            return
        }
        
        c.Next()
    }
}

// 在路由中使用
func Init(r *mvc.Router) {
    // 登录接口限制较松
    r.POST("/login", RateLimitMiddleware(10, 300))  // 5分钟内最多10次
    
    // 普通接口限制适中
    r.GET("/api/data", RateLimitMiddleware(100, 60))  // 1分钟内最多100次
    
    // 敏感操作限制严格
    r.POST("/api/transfer", RateLimitMiddleware(5, 60))  // 1分钟内最多5次
}
```

### 5.2 自定义验证规则

```go
// 添加自定义验证规则
func init() {
    validator.AddRule("chinese_phone", func(value interface{}) bool {
        phone, ok := value.(string)
        if !ok {
            return false
        }
        matched, _ := regexp.MatchString(`^1[3-9]\d{9}$`, phone)
        return matched
    })
    
    validator.AddRule("safe_string", func(value interface{}) bool {
        str, ok := value.(string)
        if !ok {
            return false
        }
        // 检查是否包含潜在危险字符
        dangerousChars := []string{"<", ">", "'", "\"", ";", "--", "/*", "*/", "("}
        for _, char := range dangerousChars {
            if strings.Contains(str, char) {
                return false
            }
        }
        return true
    })
}

// 使用自定义验证规则
type CommentValidator struct {
    validator.BaseValidator
}

func (v *CommentValidator) Rules() map[string]string {
    return map[string]string{
        "content": "required|safe_string|max:1000",
        "phone":   "required|chinese_phone",
    }
}
```

## 6. 认证与授权

### 6.1 JWT认证

```go
// JWT中间件配置
func JWTAuthMiddleware() mvc.HandlerFunc {
    return func(c *mvc.Context) {
        tokenString := c.GetHeader("Authorization")
        
        if tokenString == "" {
            c.Error(401, "Authorization header required")
            c.Abort()
            return
        }
        
        // 移除 "Bearer " 前缀
        if strings.HasPrefix(tokenString, "Bearer ") {
            tokenString = tokenString[7:]
        }
        
        // 验证JWT令牌
        claims, err := middleware.VerifyJWT(tokenString)
        if err != nil {
            c.Error(401, "Invalid token")
            c.Abort()
            return
        }
        
        // 将用户信息存储到上下文中
        c.Set("user_id", claims.UserID)
        c.Set("user_role", claims.Role)
        
        c.Next()
    }
}

// 在需要认证的路由中使用
func Init(r *mvc.Router) {
    protected := r.Group("/api")
    protected.Use(JWTAuthMiddleware())
    
    protected.GET("/dashboard", DashboardHandler)
    protected.POST("/settings", UpdateSettingsHandler)
}
```

### 6.2 RBAC权限控制

```go
// 权限检查中间件
func PermissionMiddleware(requiredPermission string) mvc.HandlerFunc {
    return func(c *mvc.Context) {
        userID := c.GetInt("user_id")
        userRole := c.GetString("user_role")
        
        // 检查用户权限
        if !hasPermission(userID, userRole, requiredPermission) {
            c.Error(403, "Insufficient permissions")
            c.Abort()
            return
        }
        
        c.Next()
    }
}

// 权限检查逻辑
func hasPermission(userID int, role, permission string) bool {
    // 管理员拥有所有权限
    if role == "admin" {
        return true
    }
    
    // 从数据库或缓存中检查用户权限
    userPermissions, err := db.Table("user_permissions").
        Where("user_id", userID).
        Pluck("permission")
    
    if err != nil {
        return false
    }
    
    for _, perm := range userPermissions {
        if perm == permission {
            return true
        }
    }
    
    return false
}

// 使用权限控制
func Init(r *mvc.Router) {
    r.GET("/admin/users", JWTAuthMiddleware(), PermissionMiddleware("user.view"), ListUsersHandler)
    r.POST("/admin/users", JWTAuthMiddleware(), PermissionMiddleware("user.create"), CreateUserHandler)
    r.PUT("/admin/users/:id", JWTAuthMiddleware(), PermissionMiddleware("user.update"), UpdateUserHandler)
    r.DELETE("/admin/users/:id", JWTAuthMiddleware(), PermissionMiddleware("user.delete"), DeleteUserHandler)
}
```

## 7. 安全日志记录

### 7.1 安全事件日志

```go
// 安全日志记录器
type SecurityLogger struct {
    logger *log.Logger
}

func NewSecurityLogger() *SecurityLogger {
    return &SecurityLogger{
        logger: log.New("security"),
    }
}

func (s *SecurityLogger) LogSecurityEvent(eventType, message string, c *mvc.Context) {
    logEntry := map[string]interface{}{
        "timestamp": time.Now().Format(time.RFC3339),
        "event_type": eventType,
        "message": message,
        "ip": c.ClientIP(),
        "user_agent": c.GetHeader("User-Agent"),
        "method": c.Method(),
        "path": c.Request.URL.Path,
        "user_id": c.GetInt("user_id"), // 如果存在
    }
    
    s.logger.Info(logEntry)
}

var securityLogger = NewSecurityLogger()

// 在登录尝试时记录
func LoginHandler(c *mvc.Context) {
    username := c.PostForm("username")
    password := c.PostForm("password")
    
    // 验证凭据
    user, err := authenticateUser(username, password)
    if err != nil {
        // 记录失败的登录尝试
        securityLogger.LogSecurityEvent("login_failed", 
            fmt.Sprintf("Failed login attempt for user: %s", username), c)
        
        c.Error(401, "Invalid credentials")
        return
    }
    
    // 记录成功的登录
    securityLogger.LogSecurityEvent("login_success", 
        fmt.Sprintf("Successful login for user: %s", username), c)
    
    // 生成JWT令牌
    token, err := middleware.GenerateJWT(user.ID, user.Role)
    if err != nil {
        c.Error(500, "Token generation failed")
        return
    }
    
    c.Json(200, gin.H{"token": token})
}
```

### 7.2 敏感操作审计

```go
// 敏感操作审计
func AuditSensitiveOperation(operation string, userID int, details map[string]interface{}) {
    auditLog := map[string]interface{}{
        "timestamp": time.Now().Format(time.RFC3339),
        "operation": operation,
        "user_id": userID,
        "details": details,
        "ip": c.ClientIP(), // 从上下文获取
    }
    
    // 记录到专门的审计日志
    auditLogger := log.New("audit")
    auditLogger.Info(auditLog)
    
    // 对于高风险操作，发送告警
    if isHighRiskOperation(operation) {
        sendAlert(fmt.Sprintf("High-risk operation detected: %s by user %d", operation, userID))
    }
}

// 检查是否为高风险操作
func isHighRiskOperation(operation string) bool {
    highRiskOps := []string{
        "user.delete",      // 删除用户
        "account.transfer", // 资金转账
        "permission.grant", // 授予权限
        "config.modify",    // 修改配置
        "data.export",      // 数据导出
    }
    
    for _, op := range highRiskOps {
        if operation == op {
            return true
        }
    }
    return false
}

// 在敏感操作中使用
func DeleteUserHandler(c *mvc.Context) {
    userID := c.GetInt("user_id")
    targetUserID := c.ParamInt("id")
    
    // 记录删除操作审计
    AuditSensitiveOperation("user.delete", userID, map[string]interface{}{
        "target_user_id": targetUserID,
        "reason": c.PostForm("reason"),
    })
    
    // 执行删除操作
    affected, err := db.Table("users").Where("id", targetUserID).Delete()
    if err != nil {
        c.Error(500, "Delete failed")
        return
    }
    
    c.Json(200, gin.H{"deleted": affected})
}
```

## 8. HTTPS和安全头

### 8.1 强制HTTPS

```go
// HTTPS强制中间件
func ForceHTTPS() mvc.HandlerFunc {
    return func(c *mvc.Context) {
        if c.Request.TLS == nil {
            // 重定向到HTTPS
            httpsURL := "https://" + c.Request.Host + c.Request.URL.RequestURI()
            c.Redirect(301, httpsURL)
            return
        }
        c.Next()
    }
}

// 在路由中使用
func Init(r *mvc.Router) {
    if config.AppEnv() == "production" {
        r.Use(ForceHTTPS())
    }
}
```

### 8.2 安全HTTP头设置

```go
// 安全头中间件
func SecurityHeaders() mvc.HandlerFunc {
    return func(c *mvc.Context) {
        // X-Frame-Options: 防止点击劫持
        c.Header("X-Frame-Options", "DENY")
        
        // X-Content-Type-Options: 防止MIME类型嗅探
        c.Header("X-Content-Type-Options", "nosniff")
        
        // X-XSS-Protection: 启用浏览器XSS防护
        c.Header("X-XSS-Protection", "1; mode=block")
        
        // Strict-Transport-Security: 强制HTTPS
        c.Header("Strict-Transport-Security", "max-age=31536000; includeSubDomains")
        
        // Content-Security-Policy: 防止各种注入攻击
        c.Header("Content-Security-Policy", 
            "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'; img-src 'self' data: https:")
        
        // Referrer-Policy: 控制Referer头
        c.Header("Referrer-Policy", "strict-origin-when-cross-origin")
        
        c.Next()
    }
}

// 在应用启动时启用
func main() {
    app := mvc.New()
    app.Use(SecurityHeaders())
    // ... 其他初始化代码
    app.Run()
}
```
