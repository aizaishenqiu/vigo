## 二、核心目录详解

### 2.1 internal/ - 私有包

**作用**: 存放服务内部实现代码，不能被其他服务直接引用

#### handler/ - HTTP/gRPC 处理器

```go
// internal/handler/user_handler.go
package handler

import (
    "net/http"
    "strconv"

    "user-service/internal/service"
    "user-service/internal/model"

    "vigo/framework/mvc"
)

// UserHandler 用户处理器
type UserHandler struct {
    userService *service.UserService
}

// NewUserHandler 创建用户处理器
func NewUserHandler() *UserHandler {
    return &UserHandler{
        userService: service.NewUserService(),
    }
}

// List 用户列表
func (h *UserHandler) List(c *mvc.Context) {
    page, _ := strconv.Atoi(c.Query("page", "1"))
    pageSize, _ := strconv.Atoi(c.Query("page_size", "10"))

    users, total, err := h.userService.GetList(c.Request.Context(), page, pageSize)
    if err != nil {
        c.Json(http.StatusInternalServerError, map[string]interface{}{
            "code":    500,
            "message": err.Error(),
        })
        return
    }

    c.Json(http.StatusOK, map[string]interface{}{
        "code": 200,
        "data": map[string]interface{}{
            "list":  users,
            "total": total,
        },
    })
}

// Create 创建用户
func (h *UserHandler) Create(c *mvc.Context) {
    var req model.CreateUserRequest
    if err := c.Bind(&req); err != nil {
        c.Json(http.StatusBadRequest, map[string]interface{}{
            "code":    400,
            "message": "参数错误",
        })
        return
    }

    user, err := h.userService.Create(c.Request.Context(), &req)
    if err != nil {
        c.Json(http.StatusInternalServerError, map[string]interface{}{
            "code":    500,
            "message": err.Error(),
        })
        return
    }

    c.Json(http.StatusCreated, map[string]interface{}{
        "code": 201,
        "data": user,
    })
}
```

#### service/ - 业务逻辑层

```go
// internal/service/user_service.go
package service

import (
    "context"
    "errors"
    "time"

    "user-service/internal/model"
    "user-service/internal/repository"
    "user-service/internal/pkg/validator"

    "vigo/framework/cache"
    "vigo/framework/log"
)

// UserService 用户服务
type UserService struct {
    userRepo *repository.UserRepository
    cache    cache.Cache
}

// NewUserService 创建用户服务
func NewUserService() *UserService {
    return &UserService{
        userRepo: repository.NewUserRepository(),
        cache:    cache.NewRedisCache(),
    }
}

// GetList 获取用户列表
func (s *UserService) GetList(ctx context.Context, page, pageSize int) ([]*model.User, int, error) {
    return s.userRepo.FindPage(ctx, page, pageSize)
}

// Create 创建用户
func (s *UserService) Create(ctx context.Context, req *model.CreateUserRequest) (*model.User, error) {
    // 参数验证
    if err := validator.Validate(req); err != nil {
        return nil, err
    }

    // 检查邮箱是否已存在
    exists, err := s.userRepo.ExistsByEmail(ctx, req.Email)
    if err != nil {
        return nil, err
    }
    if exists {
        return nil, errors.New("邮箱已存在")
    }

    // 创建用户
    user := &model.User{
        Email:     req.Email,
        Password:  req.Password,
        CreatedAt: time.Now(),
    }

    if err := user.HashPassword(); err != nil {
        return nil, err
    }

    if err := s.userRepo.Create(ctx, user); err != nil {
        return nil, err
    }

    // 清理缓存
    s.cache.Del("user:list:*")

    log.Log.Infof("用户创建成功：%d", user.ID)
    return user, nil
}
```

#### repository/ - 数据访问层

```go
// internal/repository/user_repository.go
package repository

import (
    "context"
    "database/sql"
    "time"

    "user-service/internal/model"

    "vigo/framework/db"
)

// UserRepository 用户数据访问
type UserRepository struct {
    db *db.DB
}

// NewUserRepository 创建用户数据访问
func NewUserRepository() *UserRepository {
    return &UserRepository{
        db: db.GetDB(),
    }
}

// FindPage 分页查询用户
func (r *UserRepository) FindPage(ctx context.Context, page, pageSize int) ([]*model.User, int, error) {
    offset := (page - 1) * pageSize

    // 查询总数
    var total int
    err := r.db.Table("users").Count(&total).Error
    if err != nil {
        return nil, 0, err
    }

    // 查询数据
    var users []*model.User
    err = r.db.Table("users").
        Offset(offset).
        Limit(pageSize).
        Order("id DESC").
        Find(&users).Error
    if err != nil {
        return nil, 0, err
    }

    return users, total, nil
}

// Create 创建用户
func (r *UserRepository) Create(ctx context.Context, user *model.User) error {
    return r.db.Table("users").Create(user).Error
}

// ExistsByEmail 检查邮箱是否存在
func (r *UserRepository) ExistsByEmail(ctx context.Context, email string) (bool, error) {
    var count int64
    err := r.db.Table("users").
        Where("email = ?", email).
        Count(&count).Error
    return count > 0, err
}
```

```go
    GetUser(context.Context, *GetUserRequest) (*GetUserResponse, error)
```

#### client/ - 客户端 SDK

```go
// pkg/client/user_client.go
package client

import (
    "context"
    "time"

    pb "user-service/pkg/api/v1"

    "google.golang.org/grpc"
    "google.golang.org/grpc/credentials/insecure"
)

// UserClient 用户服务客户端
type UserClient struct {
    conn   *grpc.ClientConn
    client pb.UserServiceClient
}

// NewUserClient 创建用户服务客户端
func NewUserClient(addr string) (*UserClient, error) {
    conn, err := grpc.Dial(addr,
        grpc.WithTransportCredentials(insecure.NewCredentials()),
        grpc.WithDefaultCallOptions(
            grpc.MaxCallRecvMsgSize(4*1024*1024),
            grpc.MaxCallSendMsgSize(4*1024*1024),
        ),
        grpc.WithDefaultCallOptions(grpc.WaitForReady(true)),
        grpc.WithTimeout(5*time.Second),
    )
    if err != nil {
        return nil, err
    }

    return &UserClient{
        conn:   conn,
        client: pb.NewUserServiceClient(conn),
    }, nil
}

// Close 关闭连接
func (c *UserClient) Close() error {
    return c.conn.Close()
}

// GetUser 获取用户
func (c *UserClient) GetUser(ctx context.Context, userID int64) (*pb.GetUserResponse, error) {
    return c.client.GetUser(ctx, &pb.GetUserRequest{Id: userID})
}

// CreateUser 创建用户
func (c *UserClient) CreateUser(ctx context.Context, req *pb.CreateUserRequest) (*pb.CreateUserResponse, error) {
    return c.client.CreateUser(ctx, req)
}
```

## 五、服务间通信

### 5.1 HTTP 通信

```go
// internal/service/order_service.go
package service

import (
    "context"
    "encoding/json"
    "net/http"
    "time"
)

// UserDTO 用户数据传输对象
type UserDTO struct {
    ID       int64  `json:"id"`
    Email    string `json:"email"`
    Username string `json:"username"`
}

// getUserFromService 从用户服务获取用户信息
func (s *OrderService) getUserFromService(ctx context.Context, userID int64) (*UserDTO, error) {
    // 创建 HTTP 客户端
    client := &http.Client{
        Timeout: 5 * time.Second,
    }

    // 发送请求
    req, err := http.NewRequestWithContext(ctx, "GET",
        fmt.Sprintf("http://user-service:8080/api/users/%d", userID), nil)
    if err != nil {
        return nil, err
    }

    resp, err := client.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("用户服务返回错误：%d", resp.StatusCode)
    }

    var result struct {
        Code int      `json:"code"`
        Data *UserDTO `json:"data"`
    }

    if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
        return nil, err
    }

    return result.Data, nil
}
```

## 六、最佳实践

### 6.1 目录命名规范

- ✅ 使用小写字母和下划线：`user_service/`
- ✅ 使用连字符：`user-service/`
- ❌ 避免大写字母：`UserService/`
- ❌ 避免中文：`用户服务/`

### 6.2 代码组织原则

1. **单一职责**: 每个目录只负责一个功能领域
2. **高内聚低耦合**: 相关代码放在一起，减少依赖
3. **接口抽象**: 通过接口定义层间依赖
4. **依赖注入**: 使用依赖注入容器管理依赖关系

### 6.3 配置管理

1. **环境分离**: 不同环境使用不同配置文件
2. **敏感信息**: 使用环境变量或配置中心管理
3. **配置验证**: 启动时验证配置有效性
4. **热更新**: 支持配置动态更新

### 6.4 日志规范

```go
// 结构化日志
log.Log.WithFields(log.Fields{
    "user_id": userID,
    "action":  "create",
}).Info("用户创建成功")

// 错误日志
log.Log.WithError(err).WithFields(log.Fields{
    "user_id": userID,
}).Error("用户创建失败")
```

### 6.5 错误处理

```go
// 定义错误类型
var (
    ErrUserNotFound     = errors.New("用户不存在")
    ErrUserAlreadyExist = errors.New("用户已存在")
)

// 使用错误包装
if err := s.userRepo.Create(ctx, user); err != nil {
    return nil, fmt.Errorf("创建用户失败：%w", err)
}

// 错误判断
if errors.Is(err, ErrUserNotFound) {
    // 处理用户不存在的情况
}
```
