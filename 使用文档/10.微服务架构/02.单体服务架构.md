# 单体服务架构

## 简介

单体服务架构是一种将所有功能模块打包在一个应用中的架构风格。适合中小型项目，开发简单、部署方便、调试容易。

## 单体服务实现

### 应用入口

```go
// main.go
package main

import (
    "log"
    "net/http"

    "myapp/config"
    "myapp/routes"
    "myapp/framework/db"
    "myapp/framework/cache"
    "myapp/framework/log"
)

func main() {
    // 加载配置
    cfg, err := config.Load("config/config.yaml")
    if err != nil {
        log.Fatal("加载配置失败:", err)
    }

    // 初始化数据库
    err = db.Init("mysql", cfg.Database.DSN,
        cfg.Database.MaxOpen,
        cfg.Database.MaxIdle,
        cfg.Database.MaxLifeTime)
    if err != nil {
        log.Fatal("数据库连接失败:", err)
    }

    // 初始化缓存
    cache.Init(cfg.Redis.Addr, cfg.Redis.Password, cfg.Redis.DB)

    // 初始化日志
    log.Init(cfg.Log.Level, cfg.Log.Path)

    // 注册路由
    router := routes.RegisterRoutes()

    // 启动服务
    log.Printf("服务启动在 %s", cfg.Server.Addr)
    if err := http.ListenAndServe(cfg.Server.Addr, router); err != nil {
        log.Fatal("服务启动失败:", err)
    }
}
```

### 路由定义

```go
// routes/routes.go
package routes

import (
    "net/http"

    "myapp/app/controller"
    "myapp/app/middleware"
)

func RegisterRoutes() *http.ServeMux {
    mux := http.NewServeMux()

    // 静态文件
    mux.Handle("/static/", http.StripPrefix("/static/", http.FileServer(http.Dir("static"))))

    // 首页
    mux.HandleFunc("/", controller.Home)

    // 用户路由
    mux.HandleFunc("/users", middleware.Logger(controller.UserList))
    mux.HandleFunc("/users/", middleware.Logger(controller.UserDetail))
    mux.HandleFunc("/users/create", middleware.Logger(controller.UserCreate))

    // 订单路由（需要认证）
    mux.HandleFunc("/orders", middleware.Auth(controller.OrderList))
    mux.HandleFunc("/orders/", middleware.Auth(controller.OrderDetail))
    mux.HandleFunc("/orders/create", middleware.Auth(controller.OrderCreate))

    // API路由
    mux.HandleFunc("/api/users", controller.UserAPI)
    mux.HandleFunc("/api/orders", middleware.Auth(controller.OrderAPI))

    return mux
}
```

### 控制器实现

```go
// app/controller/user.go
package controller

import (
    "encoding/json"
    "net/http"
    "strconv"

    "myapp/app/model"
    "myapp/app/service"
    "myapp/framework/db"
)

// UserList 用户列表页面
func UserList(w http.ResponseWriter, r *http.Request) {
    page, _ := strconv.Atoi(r.URL.Query().Get("page"))
    if page == 0 {
        page = 1
    }

    userService := service.NewUserService()
    users, total, err := userService.GetList(page, 10)
    if err != nil {
        http.Error(w, err.Error(), http.StatusInternalServerError)
        return
    }

    data := map[string]interface{}{
        "Title": "用户列表",
        "Users": users,
        "Pagination": map[string]interface{}{
            "Page":  page,
            "Total": total,
        },
    }

    renderTemplate(w, "users/index.html", data)
}

// UserDetail 用户详情页面
func UserDetail(w http.ResponseWriter, r *http.Request) {
    idStr := r.URL.Path[len("/users/"):]
    id, err := strconv.ParseInt(idStr, 10, 64)
    if err != nil {
        http.Error(w, "无效的用户ID", http.StatusBadRequest)
        return
    }

    userService := service.NewUserService()
    user, err := userService.GetByID(id)
    if err != nil {
        http.Error(w, err.Error(), http.StatusInternalServerError)
        return
    }

    if user == nil {
        http.Error(w, "用户不存在", http.StatusNotFound)
        return
    }

    data := map[string]interface{}{
        "Title": "用户详情",
        "User":  user,
    }

    renderTemplate(w, "users/detail.html", data)
}

// UserAPI 用户API接口
func UserAPI(w http.ResponseWriter, r *http.Request) {
    w.Header().Set("Content-Type", "application/json")

    switch r.Method {
    case "GET":
        userListAPI(w, r)
    case "POST":
        userCreateAPI(w, r)
    default:
        http.Error(w, "Method Not Allowed", http.StatusMethodNotAllowed)
    }
}

func userListAPI(w http.ResponseWriter, r *http.Request) {
    page, _ := strconv.Atoi(r.URL.Query().Get("page"))
    if page == 0 {
        page = 1
    }

    userService := service.NewUserService()
    users, total, err := userService.GetList(page, 10)
    if err != nil {
        json.NewEncoder(w).Encode(map[string]interface{}{
            "code":    500,
            "message": err.Error(),
        })
        return
    }

    json.NewEncoder(w).Encode(map[string]interface{}{
        "code":    0,
        "message": "success",
        "data": map[string]interface{}{
            "list":  users,
            "total": total,
        },
    })
}
```

### 服务层实现

```go
// app/service/user.go
package service

import (
    "myapp/app/model"
    "myapp/framework/db"
    "myapp/framework/cache"
)

// UserService 用户服务
type UserService struct {
    user *model.User
}

func NewUserService() *UserService {
    return &UserService{
        user: &model.User{},
    }
}

// GetByID 根据ID获取用户
func (us *UserService) GetByID(id int64) (*model.User, error) {
    // 先查缓存
    cacheKey := fmt.Sprintf("user:%d", id)
    if cached, err := cache.Get(cacheKey); err == nil {
        var user model.User
        if err := json.Unmarshal([]byte(cached), &user); err == nil {
            return &user, nil
        }
    }

    // 查数据库
    user, err := us.user.FindByID(id)
    if err != nil {
        return nil, err
    }

    // 写入缓存
    if user != nil {
        data, _ := json.Marshal(user)
        cache.Set(cacheKey, string(data), 3600)
    }

    return user, nil
}

// GetList 获取用户列表
func (us *UserService) GetList(page, pageSize int) ([]*model.User, int64, error) {
    return us.user.GetList(page, pageSize)
}

// Create 创建用户
func (us *UserService) Create(user *model.User) (int64, error) {
    // 验证
    if err := user.Validate(); err != nil {
        return 0, err
    }

    // 密码加密
    user.Password = hashPassword(user.Password)

    // 创建
    id, err := user.Create()
    if err != nil {
        return 0, err
    }

    return id, nil
}
```

### 中间件实现

```go
// app/middleware/auth.go
package middleware

import (
    "context"
    "net/http"
    "strings"

    "myapp/framework/jwt"
)

// Auth 认证中间件
func Auth(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("Authorization")
        if token == "" {
            http.Error(w, "未授权访问", http.StatusUnauthorized)
            return
        }

        token = strings.TrimPrefix(token, "Bearer ")
        claims, err := jwt.ParseToken(token)
        if err != nil {
            http.Error(w, "无效的token", http.StatusUnauthorized)
            return
        }

        // 将用户信息存入上下文
        ctx := context.WithValue(r.Context(), "user_id", claims.UserID)
        ctx = context.WithValue(ctx, "username", claims.Username)

        next(w, r.WithContext(ctx))
    }
}

// Logger 日志中间件
func Logger(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()

        // 调用下一个处理器
        next(w, r)

        // 记录日志
        log.Printf("[%s] %s %v", r.Method, r.URL.Path, time.Since(start))
    }
}
```

## 模块化设计

### 模块划分

```go
// app/module/user/module.go
package user

import (
    "net/http"
)

// UserModule 用户模块
type UserModule struct {
    controller *UserController
    service    *UserService
}

func NewUserModule() *UserModule {
    return &UserModule{
        controller: NewUserController(),
        service:    NewUserService(),
    }
}

// RegisterRoutes 注册模块路由
func (m *UserModule) RegisterRoutes(mux *http.ServeMux) {
    mux.HandleFunc("/users", m.controller.List)
    mux.HandleFunc("/users/", m.controller.Detail)
    mux.HandleFunc("/users/create", m.controller.Create)
}
```

```go
// main.go 模块注册
func main() {
    // ... 初始化代码 ...

    mux := http.NewServeMux()

    // 注册各模块
    userModule := user.NewUserModule()
    userModule.RegisterRoutes(mux)

    orderModule := order.NewOrderModule()
    orderModule.RegisterRoutes(mux)

    productModule := product.NewProductModule()
    productModule.RegisterRoutes(mux)

    http.ListenAndServe(":8080", mux)
}
```

### Docker部署

```dockerfile

FROM golang:1.21-alpine AS builder

WORKDIR /app
COPY . .
RUN go mod download
RUN go build -o myapp main.go

FROM alpine:latest

WORKDIR /app
COPY --from=builder /app/myapp .
COPY --from=builder /app/config ./config
COPY --from=builder /app/views ./views
COPY --from=builder /app/static ./static

EXPOSE 8080
CMD ["./myapp"]
```


## 注意事项

1. 单体服务适合中小型项目，开发效率高
2. 模块化设计便于后期拆分为微服务
3. 合理使用缓存提升性能
4. 数据库连接池配置要合理
5. 日志和监控不可缺少
6. 静态资源建议使用CDN
7. 配置文件外置，便于不同环境部署
8. 做好代码分层，保持代码可维护性
