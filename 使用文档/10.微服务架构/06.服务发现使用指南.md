# Vigo 框架 - 服务发现使用指南

## 1. 概述

Vigo 框架集成了 Nacos 服务发现组件，提供服务注册、服务发现、健康检查等功能，支持微服务架构中的服务治理。

### 特性

- ✅ Nacos 服务注册
- ✅ 动态服务发现
- ✅ 健康检查与故障剔除
- ✅ 服务权重配置
- ✅ 元数据管理
- ✅ 多命名空间支持
- ✅ 服务实例变更监听

## 2. 快速开始

### 2.1 安装依赖

```bash
go get github.com/nacos-group/nacos-sdk-go/v2
```

### 2.2 初始化服务发现

```go
package main

import (
    "context"
    "log"
    "net"
    "os"
    "os/signal"
    "syscall"
    
    "github.com/yourorg/vigo/framework/discovery"
    "github.com/yourorg/vigo/framework/app"
)

func main() {
    // 创建服务发现客户端
    disc, err := discovery.NewDiscovery(
        discovery.WithServerAddr("http://localhost:8848"),
        discovery.WithNamespace("public"),
        discovery.WithGroup("DEFAULT_GROUP"),
    )
    if err != nil {
        log.Fatalf("Failed to create discovery client: %v", err)
    }

    // 获取本机 IP
    ip, err := getLocalIP()
    if err != nil {
        log.Fatalf("Failed to get local IP: %v", err)
    }

    // 注册服务实例
    instance := &discovery.Instance{
        IP:       ip,
        Port:     8080,
        Weight:   1.0,
        Enabled:  true,
        Healthy:  true,
        Metadata: map[string]string{
            "version": "1.0.0",
            "region":  "cn-beijing",
        },
    }

    ctx := context.Background()
    err = disc.Register(ctx, "user-service", instance)
    if err != nil {
        log.Fatalf("Failed to register service: %v", err)
    }

    // 启动应用
    go app.Run()

    // 监听退出信号
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit

    // 优雅停机：注销服务实例
    if err := disc.Deregister(ctx, "user-service", instance); err != nil {
        log.Printf("Failed to deregister service: %v", err)
    }

    log.Println("Service stopped gracefully")
}

// 获取本机 IP
func getLocalIP() (string, error) {
    addrs, err := net.InterfaceAddrs()
    if err != nil {
        return "", err
    }

    for _, addr := range addrs {
        if ipnet, ok := addr.(*net.IPNet); ok && !ipnet.IP.IsLoopback() {
            if ipnet.IP.To4() != nil {
                return ipnet.IP.String(), nil
            }
        }
    }
    return "", fmt.Errorf("no valid IP found")
}
```

```go
        }
    }()

    return nil
```

## 5. 集成负载均衡

### 5.1 创建负载均衡器

```go
import (
    "github.com/yourorg/vigo/framework/discovery"
    "github.com/yourorg/vigo/framework/balancer"
)

func CreateLoadBalancer() error {
    disc, _ := discovery.NewDiscovery(
        discovery.WithServerAddr("http://localhost:8848"),
    )

    ctx := context.Background()
    
    // 获取服务实例
    instances, err := disc.GetHealthyInstances(ctx, "user-service")
    if err != nil {
        return err
    }

    // 转换为负载均衡器实例
    lbInstances := make([]*balancer.Instance, len(instances))
    for i, inst := range instances {
        lbInstances[i] = &balancer.Instance{
            Address: fmt.Sprintf("%s:%d", inst.IP, inst.Port),
            Weight:  inst.Weight,
            Metadata: inst.Metadata,
        }
    }

    // 创建轮询负载均衡器
    balancer := balancer.CreateBalancer(
        balancer.RoundRobin,
        lbInstances,
    )

    // 获取下一个实例
    nextInst, err := balancer.GetInstance(ctx)
    if err != nil {
        return err
    }

    log.Printf("Selected instance: %s", nextInst.Address)
    return nil
}
```

```go
        }
    }()

    return nil
}
```

## 6. HTTP 客户端集成

### 6.1 服务发现 HTTP 客户端

```go
import (
    "context"
    "net/http"
    "net/http/httputil"
    "github.com/yourorg/vigo/framework/discovery"
    "github.com/yourorg/vigo/framework/balancer"
)

type ServiceDiscoveryClient struct {
    discovery  *discovery.Discovery
    balancer   balancer.LoadBalancer
    httpClient *http.Client
}

func NewServiceDiscoveryClient(serverAddr string) (*ServiceDiscoveryClient, error) {
    disc, err := discovery.NewDiscovery(
        discovery.WithServerAddr(serverAddr),
    )
    if err != nil {
        return nil, err
    }

    return &ServiceDiscoveryClient{
        discovery:  disc,
        httpClient: &http.Client{Timeout: 30 * time.Second},
    }, nil
}

func (c *ServiceDiscoveryClient) CallService(ctx context.Context, serviceName, path string) (*http.Response, error) {
    // 获取实例
    inst, err := c.balancer.GetInstance(ctx)
    if err != nil {
        return nil, err
    }

    // 创建请求
    url := fmt.Sprintf("http://%s%s", inst.Address, path)
    req, err := http.NewRequestWithContext(ctx, "GET", url, nil)
    if err != nil {
        return nil, err
    }

    // 发送请求
    return c.httpClient.Do(req)
}
```

### 6.2 带重试的 HTTP 调用

```go
func (c *ServiceDiscoveryClient) CallWithRetry(ctx context.Context, serviceName, path string, maxRetries int) (*http.Response, error) {
    var lastErr error
    
    for i := 0; i < maxRetries; i++ {
        // 获取实例
        inst, err := c.balancer.GetInstance(ctx)
        if err != nil {
            lastErr = err
            continue
        }

        // 创建请求
        url := fmt.Sprintf("http://%s%s", inst.Address, path)
        req, _ := http.NewRequestWithContext(ctx, "GET", url, nil)

        // 发送请求
        resp, err := c.httpClient.Do(req)
        if err == nil && resp.StatusCode < 500 {
            return resp, nil
        }

        if err != nil {
            lastErr = err
        } else {
            lastErr = fmt.Errorf("status: %d", resp.StatusCode)
            resp.Body.Close()
        }

        // 标记实例为不健康（可选）
        c.balancer.MarkUnhealthy(inst)
    }

    return nil, fmt.Errorf("all retries failed: %v", lastErr)
}
```

## 8. 健康检查

### 8.1 主动健康检查

```go
func HealthCheck(ctx context.Context, disc *discovery.Discovery, serviceName string) error {
    instances, err := disc.GetInstances(ctx, serviceName)
    if err != nil {
        return err
    }

    for _, inst := range instances {
        // HTTP 健康检查
        url := fmt.Sprintf("http://%s:%d/health", inst.IP, inst.Port)
        resp, err := http.Get(url)
        
        healthy := err == nil && resp.StatusCode == 200
        
        if healthy != inst.Healthy {
            log.Printf("Instance %s:%d health status changed: %v -> %v",
                inst.IP, inst.Port, inst.Healthy, healthy)
            
            // 更新实例健康状态
            inst.Healthy = healthy
            if err := disc.Register(ctx, serviceName, inst); err != nil {
                log.Printf("Failed to update instance health: %v", err)
            }
        }
    }

    return nil
}
```

## 10. 灰度发布

### 10.1 基于元数据的灰度路由

```go
func GrayReleaseRouting(ctx context.Context, disc *discovery.Discovery, userID string) (*discovery.Instance, error) {
    // 获取所有实例
    instances, err := disc.GetInstances(ctx, "user-service")
    if err != nil {
        return nil, err
    }

    // 判断用户是否为灰度用户
    isGrayUser := isGrayUser(userID)

    // 过滤实例
    var targetInstances []*discovery.Instance
    for _, inst := range instances {
        if isGrayUser && inst.Metadata["gray"] == "true" {
            targetInstances = append(targetInstances, inst)
        } else if !isGrayUser && inst.Metadata["gray"] != "true" {
            targetInstances = append(targetInstances, inst)
        }
    }

    if len(targetInstances) == 0 {
        return nil, errors.New("no available instances")
    }

    // 随机选择一个实例
    return targetInstances[rand.Intn(len(targetInstances))], nil
}

func isGrayUser(userID string) bool {
    // 灰度用户判断逻辑（如：按用户 ID 哈希）
    hash := fnv.New32a().Sum32()
    return hash%100 < 10 // 10% 的用户为灰度用户
}
```

## 12. 常见问题

### Q1: 服务注册失败？

**检查项：**
1. Nacos 服务器是否正常运行
2. 网络是否可达
3. 命名空间和分组是否正确
4. 实例 IP 和端口是否有效

### Q2: 无法发现服务实例？

**排查方法：**
1. 确认服务已注册
2. 检查命名空间是否匹配
3. 验证服务名称是否正确
4. 查看 Nacos 控制台确认实例状态

### Q3: 负载均衡不生效？

**检查项：**
1. 负载均衡器是否正确初始化
2. 实例列表是否已更新
3. 健康检查是否正常工作
4. 负载均衡策略是否正确配置

### Q4: 灰度发布不生效？

**排查方法：**
1. 验证元数据是否正确设置
2. 检查灰度用户判断逻辑
3. 确认灰度实例已注册
4. 查看路由逻辑是否正确
