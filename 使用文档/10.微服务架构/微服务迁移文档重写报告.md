```go
// 控制器层
func (h *UserHandler) List(c *mvc.Context) {
    page, _ := strconv.Atoi(c.Query("page", "1"))
    users, total, _ := h.service.GetList(c.Request.Context(), page, 10)
    c.Json(http.StatusOK, successResponse(map[string]interface{}{
        "list":  users,
        "total": total,
    }))
}

// 服务层
func (s *UserService) GetByID(ctx context.Context, id int64) (*model.User, error) {
    // 先从缓存获取
    var user model.User
    if err := s.cache.Get(fmt.Sprintf("user:%d", id), &user); err == nil {
        return &user, nil
    }
    
    // 从数据库获取
    user, err := s.repo.GetByID(ctx, id)
    if err != nil {
        return nil, err
    }
    
    // 写入缓存
    s.cache.Set(fmt.Sprintf("user:%d", id), user, 5*time.Minute)
    return user, nil
}

// 数据访问层
func (r *UserRepository) GetByID(ctx context.Context, id int64) (*model.User, error) {
    var user model.User
    err := r.db.WithContext(ctx).First(&user, id).Error
    return &user, err
}
```

### 服务间通信 - HTTP + 服务发现

```go
func (c *UserClient) GetUser(ctx context.Context, userID int64) (*User, error) {
    // 从服务发现获取地址（负载均衡）
    addr, err := discovery.GetService("user-service")
    if err != nil {
        return nil, fmt.Errorf("获取服务地址失败：%w", err)
    }

    var resp struct {
        Code int    `json:"code"`
        Data *User  `json:"data"`
    }

    url := fmt.Sprintf("%s/api/v1/users/%d", addr, userID)
    if err := c.client.Get(ctx, url, &resp); err != nil {
        return nil, fmt.Errorf("调用用户服务失败：%w", err)
    }

    if resp.Code != 200 {
        return nil, fmt.Errorf("用户服务返回错误：%d", resp.Code)
    }

    return resp.Data, nil
}
```
