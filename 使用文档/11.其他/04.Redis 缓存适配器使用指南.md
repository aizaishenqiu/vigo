# Redis 缓存适配器使用指南

## 概述

Vigo 框架提供完整的 Redis 缓存适配器，支持字符串、对象、批量操作、标签管理等功能。

## 功能特性

- ✅ 支持多种数据类型
- ✅ 批量操作
- ✅ 标签缓存管理
- ✅ 连接池配置
- ✅ 自动序列化
- ✅ TTL 过期时间
- ✅ 统计信息

## 快速开始

### 1. 安装 Redis

```bash
# Docker 安装
docker run -d \
  -p 6379:6379 \
  --name redis \
  redis:7-alpine
```

### 2. 基本使用

```go
package main

import (
    "fmt"
    "log"
    "time"
    "vigo/framework/cache"
)

func main() {
    // 创建 Redis 客户端
    client, err := cache.CreateRedisClient(
        "127.0.0.1:6379",  // 地址
        "",                // 密码
        0,                 // 数据库
        &cache.RedisCacheOptions{
            Prefix:       "app:",
            DefaultTTL:   24 * time.Hour,
            PoolSize:     100,
            DialTimeout:  5 * time.Second,
            ReadTimeout:  3 * time.Second,
            WriteTimeout: 3 * time.Second,
        },
    )
    if err != nil {
        log.Fatal(err)
    }

    // 创建缓存适配器
    redisCache := cache.NewRedisCacheAdapter(client, &cache.RedisCacheOptions{
        Prefix: "app:",
    })

    // 设置缓存
    err = redisCache.Set("user:1", map[string]interface{}{
        "id":   1,
        "name": "John",
        "age":  25,
    }, time.Hour)
    if err != nil {
        log.Fatal(err)
    }

    // 获取缓存
    value, err := redisCache.Get("user:1")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("User: %+v\n", value)

    // 删除缓存
    err = redisCache.Delete("user:1")
}
```

## 配置选项

### RedisCacheOptions

```go
type RedisCacheOptions struct {
    // 键前缀
    Prefix string `yaml:"prefix"`
    
    // 默认过期时间
    DefaultTTL time.Duration `yaml:"default_ttl"`
    
    // 最大空闲连接数
    MaxIdleConns int `yaml:"max_idle_conns"`
    
    // 连接池大小
    PoolSize int `yaml:"pool_size"`
    
    // 连接超时
    DialTimeout time.Duration `yaml:"dial_timeout"`
    
    // 读取超时
    ReadTimeout time.Duration `yaml:"read_timeout"`
    
    // 写入超时
    WriteTimeout time.Duration `yaml:"write_timeout"`
}
```

## 基本操作

### 1. 字符串操作

```go
// 设置字符串
err := redisCache.Set("name", "John", time.Hour)

// 获取字符串
name, err := redisCache.GetString("name")

// 设置并返回旧值
oldValue, err := redisCache.GetSet("name", "Jane", time.Hour)
```

### 2. 对象操作

```go
// 设置对象
user := &User{ID: 1, Name: "John"}
err := redisCache.Set("user:1", user, time.Hour)

// 获取对象
var user User
value, err := redisCache.Get("user:1")
jsonData, _ := json.Marshal(value)
json.Unmarshal(jsonData, &user)
```

### 3. 数值操作

```go
// 自增
val, err := redisCache.Incr("counter")

// 自减
val, err := redisCache.Decr("counter")

// 自增 N
val, err := redisCache.IncrBy("counter", 10)

// 自减 N
val, err := redisCache.DecrBy("counter", 5)
```

### 4. 存在性检查

```go
// 检查键是否存在
exists, err := redisCache.Exists("user:1")

// 获取剩余过期时间
ttl, err := redisCache.TTL("user:1")

// 设置过期时间
err := redisCache.Expire("user:1", time.Hour)
```

## 批量操作

## 完整示例

### 用户缓存管理

```go
package cache

import (
    "fmt"
    "time"
    "vigo/framework/cache"
)

type UserCache struct {
    cache *cache.RedisCacheAdapter
}

func NewUserCache(client *redis.Client) *UserCache {
    return &UserCache{
        cache: cache.NewRedisCacheAdapter(client, &cache.RedisCacheOptions{
            Prefix:     "user:",
            DefaultTTL: 24 * time.Hour,
        }),
    }
}

// 设置用户缓存
func (c *UserCache) SetUser(userID int64, user interface{}) error {
    key := fmt.Sprintf("user:%d", userID)
    return c.cache.Set(key, user, 24*time.Hour)
}

// 获取用户缓存
func (c *UserCache) GetUser(userID int64) (interface{}, error) {
    key := fmt.Sprintf("user:%d", userID)
    return c.cache.Get(key)
}

// 删除用户缓存
func (c *UserCache) DeleteUser(userID int64) error {
    key := fmt.Sprintf("user:%d", userID)
    return c.cache.Delete(key)
}

// 批量设置用户缓存
func (c *UserCache) SetUsers(users map[int64]interface{}) error {
    items := make(map[string]interface{})
    for id, user := range users {
        key := fmt.Sprintf("user:%d", id)
        items[key] = user
    }
    return c.cache.SetMulti(items, 24*time.Hour)
}

// 批量获取用户缓存
func (c *UserCache) GetUsers(userIDs []int64) (map[int64]interface{}, error) {
    keys := make([]string, 0, len(userIDs))
    for _, id := range userIDs {
        keys = append(keys, fmt.Sprintf("user:%d", id))
    }
    
    values, err := c.cache.GetMulti(keys)
    if err != nil {
        return nil, err
    }
    
    result := make(map[int64]interface{})
    for key, value := range values {
        // 从 key 中提取 userID
        var userID int64
        fmt.Sscanf(key, "user:%d", &userID)
        result[userID] = value
    }
    
    return result, nil
}

// 删除用户标签缓存
func (c *UserCache) DeleteUserTag(tag string) error {
    return c.cache.DeleteByTag(tag)
}

// 缓存用户并添加标签
func (c *UserCache) SetUserWithTag(userID int64, user interface{}, tags ...string) error {
    key := fmt.Sprintf("user:%d", userID)
    return c.cache.SetWithTags(key, user, 24*time.Hour, tags...)
}
```

### 分布式锁

```go
package lock

import (
    "time"
    "vigo/framework/cache"
)

type DistributedLock struct {
    cache *cache.RedisCacheAdapter
}

func NewDistributedLock(client *redis.Client) *DistributedLock {
    return &DistributedLock{
        cache: cache.NewRedisCacheAdapter(client, &cache.RedisCacheOptions{
            Prefix: "lock:",
        }),
    }
}

// 获取锁
func (l *DistributedLock) Acquire(key string, ttl time.Duration) (bool, error) {
    return l.cache.SetNX(key, "locked", ttl)
}

// 释放锁
func (l *DistributedLock) Release(key string) error {
    return l.cache.Delete(key)
}

// 尝试获取锁（带重试）
func (l *DistributedLock) TryAcquire(key string, ttl time.Duration, retry int, delay time.Duration) (bool, error) {
    for i := 0; i < retry; i++ {
        ok, err := l.cache.SetNX(key, "locked", ttl)
        if ok {
            return true, nil
        }
        if err != nil {
            return false, err
        }
        time.Sleep(delay)
    }
    return false, nil
}
```

## 最佳实践

### 1. 缓存穿透

```go
// 使用 SetNX 防止缓存穿透
func GetUserWithCache(userID int64) (*User, error) {
    key := fmt.Sprintf("user:%d", userID)
    
    // 尝试从缓存获取
    value, err := redisCache.Get(key)
    if err == nil && value != nil {
        return value.(*User), nil
    }
    
    // 缓存不存在，查询数据库
    user, err := db.GetUser(userID)
    if err != nil {
        // 数据库也不存在，设置空值防止穿透
        redisCache.SetNX(key, "", 5*time.Minute)
        return nil, err
    }
    
    // 写入缓存
    redisCache.Set(key, user, time.Hour)
    return user, nil
}
```

### 2. 缓存雪崩

```go
// 设置随机过期时间防止雪崩
func SetWithRandomTTL(key string, value interface{}, baseTTL time.Duration) error {
    // 添加随机偏移（±20%）
    offset := time.Duration(rand.Int63n(int64(baseTTL) * 0.4))
    ttl := baseTTL - baseTTL*0.2 + offset
    
    return redisCache.Set(key, value, ttl)
}
```
