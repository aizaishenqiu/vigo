# 分布式限流器使用指南

## 概述

Vigo 框架提供基于 Redis 的分布式限流器，支持令牌桶和滑动窗口两种算法，适用于高并发场景下的流量控制。

## 功能特性

- ✅ 分布式限流（基于 Redis）
- ✅ 令牌桶算法
- ✅ 滑动窗口算法
- ✅ Lua 脚本保证原子性
- ✅ 支持突发流量
- ✅ 可配置限流参数

## 快速开始

### 1. 基本使用

```go
package main

import (
    "log"
    "net/http"
    "time"
    "vigo/framework/limiter"
)

func main() {
    // 创建分布式限流器（令牌桶算法）
    limiter := limiter.NewDistributedLimiter(&limiter.DistributedLimiterOptions{
        Client:   redisClient,
        Key:      "api:limit:user:1",
        Rate:     100,  // 每秒允许 100 个请求
        Burst:    200,  // 突发容量 200
        Interval: 1 * time.Second,
    })

    // 在中间件中使用
    http.HandleFunc("/api/test", func(w http.ResponseWriter, r *http.Request) {
        if !limiter.Allow() {
            w.WriteHeader(http.StatusTooManyRequests)
            w.Write([]byte("Too Many Requests"))
            return
        }

        // 处理请求
        w.WriteHeader(http.StatusOK)
        w.Write([]byte("Success"))
    })

    log.Fatal(http.ListenAndServe(":8080", nil))
}
```

## 限流算法

### 1. 令牌桶算法

```go
// 创建令牌桶限流器
limiter := limiter.NewDistributedLimiter(&limiter.DistributedLimiterOptions{
    Client:   redisClient,
    Key:      "api:limit:token-bucket",
    Rate:     100,  // 每秒生成 100 个令牌
    Burst:    200,  // 桶容量 200 个令牌
    Interval: 1 * time.Second,
})

// 检查是否允许请求
if limiter.Allow() {
    // 处理请求
} else {
    // 拒绝请求
}
```

**工作原理**:
- 桶中有一定数量的令牌
- 每秒生成固定数量的令牌（Rate）
- 桶有最大容量（Burst）
- 每次请求消耗一个令牌
- 没有令牌时拒绝请求

**适用场景**:
- 允许突发流量
- 平滑限流
- API 限流

### 2. 滑动窗口算法

```go
// 创建滑动窗口限流器
rateLimiter := limiter.NewDistributedRateLimiter(&limiter.RateLimiterOptions{
    Client:    redisClient,
    Key:       "api:limit:sliding-window",
    Rate:      100,  // 每秒允许 100 个请求
    Precision: 100 * time.Millisecond,  // 窗口精度
    Window:    1 * time.Second,  // 窗口大小
})

if rateLimiter.Allow() {
    // 处理请求
} else {
    // 拒绝请求
}
```

**工作原理**:
- 将时间窗口分成多个小格子
- 记录每个格子的请求数
- 滑动窗口统计总请求数
- 超过限制时拒绝请求

**适用场景**:
- 精确限流
- 不允许突发
- 严格频率控制

### 1. Gin 框架

```go
import (
    "github.com/gin-gonic/gin"
    "vigo/framework/limiter"
)

// 创建限流器
userLimiter := limiter.NewDistributedLimiter(&limiter.DistributedLimiterOptions{
    Client:   redisClient,
    Key:      "api:limit:user",
    Rate:     100,
    Burst:    200,
    Interval: time.Second,
})

// 限流中间件
func RateLimitMiddleware(limiter *limiter.DistributedLimiter) gin.HandlerFunc {
    return func(c *gin.Context) {
        if !limiter.Allow() {
            c.JSON(429, gin.H{
                "error": "Too Many Requests",
            })
            c.Abort()
            return
        }
        c.Next()
    }
}

// 使用中间件
r := gin.Default()
r.Use(RateLimitMiddleware(userLimiter))
```

### 2. 按用户限流

```go
func UserRateLimitMiddleware() gin.HandlerFunc {
    limiters := make(map[string]*limiter.DistributedLimiter)
    var mu sync.RWMutex

    return func(c *gin.Context) {
        userID := c.GetHeader("X-User-ID")
        if userID == "" {
            userID = c.ClientIP()
        }

        // 获取或创建限流器
        mu.RLock()
        limiter, exists := limiters[userID]
        mu.RUnlock()

        if !exists {
            mu.Lock()
            limiter, exists = limiters[userID]
            if !exists {
                limiter = limiter.NewDistributedLimiter(&limiter.DistributedLimiterOptions{
                    Client:   redisClient,
                    Key:      "api:limit:user:" + userID,
                    Rate:     100,
                    Burst:    200,
                    Interval: time.Second,
                })
                limiters[userID] = limiter
            }
            mu.Unlock()
        }

        if !limiter.Allow() {
            c.JSON(429, gin.H{
                "error": "Too Many Requests",
            })
            c.Abort()
            return
        }

        c.Next()
    }
}
```

## 完整示例

### API 网关限流

```go
package main

import (
    "context"
    "log"
    "net/http"
    "time"
    "vigo/framework/limiter"
)

// 限流配置
type RateLimitConfig struct {
    Rate     int           `yaml:"rate"`
    Burst    int           `yaml:"burst"`
    Interval time.Duration `yaml:"interval"`
}

// API 限流器
type APILimiter struct {
    limiters map[string]*limiter.DistributedLimiter
    configs  map[string]*RateLimitConfig
    mu       sync.RWMutex
    client   *redis.Client
}

func NewAPILimiter(client *redis.Client) *APILimiter {
    return &APILimiter{
        limiters: make(map[string]*limiter.DistributedLimiter),
        configs:  make(map[string]*RateLimitConfig),
        client:   client,
    }
}

// 配置 API 限流
func (a *APILimiter) Configure(apiPath string, config *RateLimitConfig) {
    a.mu.Lock()
    defer a.mu.Unlock()
    a.configs[apiPath] = config
}

// 获取限流器
func (a *APILimiter) GetLimiter(apiPath string) *limiter.DistributedLimiter {
    a.mu.RLock()
    config, exists := a.configs[apiPath]
    a.mu.RUnlock()

    if !exists {
        // 使用默认配置
        config = &RateLimitConfig{
            Rate:     100,
            Burst:    200,
            Interval: time.Second,
        }
    }

    a.mu.Lock()
    defer a.mu.Unlock()

    limiter, exists := a.limiters[apiPath]
    if !exists {
        key := "api:limit:" + strings.ReplaceAll(apiPath, "/", ":")
        limiter = limiter.NewDistributedLimiter(&limiter.DistributedLimiterOptions{
            Client:   a.client,
            Key:      key,
            Rate:     config.Rate,
            Burst:    config.Burst,
            Interval: config.Interval,
        })
        a.limiters[apiPath] = limiter
    }

    return limiter
}

// 限流中间件
func (a *APILimiter) Middleware(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        apiPath := r.URL.Path
        limiter := a.GetLimiter(apiPath)

        if !limiter.Allow() {
            w.Header().Set("Retry-After", "1")
            w.WriteHeader(http.StatusTooManyRequests)
            w.Write([]byte("Too Many Requests"))
            return
        }

        next(w, r)
    }
}

func main() {
    // 创建 API 限流器
    apiLimiter := NewAPILimiter(redisClient)

    // 配置不同 API 的限流
    apiLimiter.Configure("/api/login", &RateLimitConfig{
        Rate:     10,   // 登录接口严格限流
        Burst:    20,
        Interval: time.Second,
    })

    apiLimiter.Configure("/api/search", &RateLimitConfig{
        Rate:     50,
        Burst:    100,
        Interval: time.Second,
    })

    apiLimiter.Configure("/api/data", &RateLimitConfig{
        Rate:     1000,  // 数据接口宽松限流
        Burst:    2000,
        Interval: time.Second,
    })

    // 使用限流中间件
    http.HandleFunc("/api/login", apiLimiter.Middleware(func(w http.ResponseWriter, r *http.Request) {
        // 登录逻辑
        w.WriteHeader(http.StatusOK)
    }))

    http.HandleFunc("/api/search", apiLimiter.Middleware(func(w http.ResponseWriter, r *http.Request) {
        // 搜索逻辑
        w.WriteHeader(http.StatusOK)
    }))

    http.HandleFunc("/api/data", apiLimiter.Middleware(func(w http.ResponseWriter, r *http.Request) {
        // 数据逻辑
        w.WriteHeader(http.StatusOK)
    }))

    log.Fatal(http.ListenAndServe(":8080", nil))
}
```

// 根据系统负载动态调整限流
func GetDynamicRateLimit() int {
    load := getSystemLoad()  // 获取系统负载
    if load > 0.8 {
        return 50  // 高负载时降低限流
    } else if load > 0.5 {
        return 100 // 中等负载
    }
    return 1000 // 正常负载
}
