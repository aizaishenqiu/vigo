# Nacos 服务发现使用指南

## 概述

Vigo 框架集成了 Nacos 服务发现，提供服务注册、服务发现、健康检查、负载均衡等能力。

## 功能特性

- ✅ 服务注册与注销
- ✅ 服务发现与查询
- ✅ 健康检查
- ✅ 负载均衡
- ✅ 服务监听
- ✅ 元数据管理
- ✅ 权重配置

## 快速开始

### 1. 安装 Nacos

```bash
# Docker 安装
docker run -d \
  -p 8848:8848 \
  -p 9848:9848 \
  -p 9849:9849 \
  --name nacos \
  -e MODE=standalone \
  nacos/nacos-server:2.3.0
```

### 2. 服务注册

```go
package main

import (
    "context"
    "log"
    "net/http"
    "time"
    "vigo/framework/discovery"
)

func main() {
    // 创建 Nacos 服务发现
    disc, err := discovery.NewNacosDiscovery(&discovery.NacosDiscoveryOptions{
        ServerAddr:  "127.0.0.1:8848",
        NamespaceId: "public",
        GroupName:   "DEFAULT_GROUP",
        ClusterName: "DEFAULT",
        TimeoutMs:   5000,
        LogDir:      "logs/nacos",
        CacheDir:    "cache/nacos",
        LogLevel:    "info",
    })
    if err != nil {
        log.Fatal(err)
    }

    // 注册服务实例
    ctx := context.Background()
    err = disc.Register(ctx, "user-service", &discovery.Instance{
        ID:       "user-service-1",
        IP:       "192.168.1.100",
        Port:     8080,
        Weight:   1.0,
        Enabled:  true,
        Healthy:  true,
        Metadata: map[string]string{
            "version": "1.0.0",
            "region":  "cn-east",
        },
    })
    if err != nil {
        log.Fatal(err)
    }

    // 启动 HTTP 服务
    http.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
        w.WriteHeader(http.StatusOK)
        w.Write([]byte("OK"))
    })

    log.Println("Service started on :8080")
    log.Fatal(http.ListenAndServe(":8080", nil))
}
```

### 3. 服务发现

```go
package main

import (
    "context"
    "fmt"
    "log"
    "vigo/framework/discovery"
)

func main() {
    // 创建服务发现客户端
    disc, err := discovery.NewNacosDiscovery(&discovery.NacosDiscoveryOptions{
        ServerAddr:  "127.0.0.1:8848",
        NamespaceId: "public",
        GroupName:   "DEFAULT_GROUP",
    })
    if err != nil {
        log.Fatal(err)
    }

    ctx := context.Background()

    // 获取服务实例列表
    instances, err := disc.GetInstances(ctx, "user-service")
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("Found %d instances:\n", len(instances))
    for _, inst := range instances {
        fmt.Printf("  - %s:%d (weight: %.2f, healthy: %v)\n",
            inst.IP, inst.Port, inst.Weight, inst.Healthy)
    }

    // 选择一个健康实例
    instance, err := disc.SelectOneHealthyInstance(ctx, "user-service")
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("Selected instance: %s:%d\n", instance.IP, instance.Port)
}
```

## 配置选项

### NacosDiscoveryOptions

```go
type NacosDiscoveryOptions struct {
    // Nacos 服务器地址
    ServerAddr string `yaml:"server_addr"`
    
    // 命名空间 ID
    NamespaceId string `yaml:"namespace_id"`
    
    // 集群名称
    ClusterName string `yaml:"cluster_name"`
    
    // 服务组
    GroupName string `yaml:"group_name"`
    
    // 超时时间（毫秒）
    TimeoutMs uint64 `yaml:"timeout_ms"`
    
    // 日志目录
    LogDir string `yaml:"log_dir"`
    
    // 缓存目录
    CacheDir string `yaml:"cache_dir"`
    
    // 日志级别
    LogLevel string `yaml:"log_level"`
    
    // 用户名（可选）
    Username string `yaml:"username"`
    
    // 密码（可选）
    Password string `yaml:"password"`
}
```

type Instance struct {

## 完整示例

### 微服务注册与发现

```go
package main

import (
    "context"
    "encoding/json"
    "log"
    "net/http"
    "time"
    "vigo/framework/discovery"
    "vigo/framework/balancer"
)

// 服务结构
type Service struct {
    discovery  *discovery.NacosDiscovery
    balancer   *balancer.Balancer
    serviceName string
}

// 创建服务
func NewService(serviceName string) (*Service, error) {
    disc, err := discovery.NewNacosDiscovery(&discovery.NacosDiscoveryOptions{
        ServerAddr:  "127.0.0.1:8848",
        NamespaceId: "public",
        GroupName:   "DEFAULT_GROUP",
    })
    if err != nil {
        return nil, err
    }

    return &Service{
        discovery:   disc,
        serviceName: serviceName,
    }, nil
}

// 注册服务
func (s *Service) Register(ip string, port int, metadata map[string]string) error {
    ctx := context.Background()
    return s.discovery.Register(ctx, s.serviceName, &discovery.Instance{
        ID:       s.serviceName + "-" + ip + ":" + string(rune(port)),
        IP:       ip,
        Port:     port,
        Weight:   1.0,
        Enabled:  true,
        Healthy:  true,
        Metadata: metadata,
    })
}

// 调用服务
func (s *Service) Call(ctx context.Context, path string) ([]byte, error) {
    // 获取健康实例
    instances, err := s.discovery.GetHealthyInstances(ctx, s.serviceName)
    if err != nil {
        return nil, err
    }

    if len(instances) == 0 {
        return nil, ErrNoInstance
    }

    // 负载均衡选择一个实例
    balancer := balancer.NewBalancer(balancer.RoundRobin)
    balancer.SetInstances(instances)
    instance := balancer.Next()

    // 发起 HTTP 请求
    url := "http://" + instance.IP + ":" + string(rune(instance.Port)) + path
    resp, err := http.Get(url)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    // 读取响应
    var result []byte
    // ... 读取 resp.Body
    
    return result, nil
}

// 启动服务监听
func (s *Service) StartWatch() error {
    ctx := context.Background()
    eventChan, err := s.discovery.Watch(ctx, s.serviceName)
    if err != nil {
        return err
    }

    go func() {
        for event := range eventChan {
            log.Printf("Service %s changed: %v", s.serviceName, event)
            
            // 更新负载均衡器
            if s.balancer != nil {
                s.balancer.SetInstances(event.Instances)
            }
        }
    }()

    return nil
}

// 优雅停机
func (s *Service) Shutdown(ctx context.Context) error {
    // 注销服务
    return s.discovery.Deregister(ctx, s.serviceName, &discovery.Instance{
        IP:   "192.168.1.100",
        Port: 8080,
    })
}

func main() {
    // 创建服务
    svc, err := NewService("user-service")
    if err != nil {
        log.Fatal(err)
    }

    // 注册服务
    err = svc.Register("192.168.1.100", 8080, map[string]string{
        "version": "1.0.0",
        "region":  "cn-east",
    })
    if err != nil {
        log.Fatal(err)
    }

    // 启动服务监听
    err = svc.StartWatch()
    if err != nil {
        log.Fatal(err)
    }

    // 启动 HTTP 服务
    http.HandleFunc("/health", func(w http.ResponseWriter, r *http.Request) {
        w.WriteHeader(http.StatusOK)
        w.Write([]byte("OK"))
    })

    log.Println("Service started on :8080")
    
    // 优雅停机
    ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    go func() {
        <-ctx.Done()
        svc.Shutdown(ctx)
    }()

    log.Fatal(http.ListenAndServe(":8080", nil))
}
```

### 2. 优雅停机

```go
func gracefulShutdown(svc *Service) {
    // 创建超时上下文
    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()

    // 等待请求处理完成
    ticker := time.NewTicker(1 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-ctx.Done():
            log.Println("Shutdown timeout")
            return
        case <-ticker.C:
            if getActiveConnections() == 0 {
                // 没有活跃连接，可以停机
                svc.Shutdown(ctx)
                return
            }
        }
    }
}
```
