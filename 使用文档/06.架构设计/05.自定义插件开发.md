# 自定义插件开发详解

Vigo 框架提供强大的插件扩展系统（Extend System），灵感来自 Laravel 的服务提供者（Service Provider）。通过插件系统，用户可以轻松扩展框架功能，实现模块化开发。

---

## 一、什么是插件系统

### 1.1 插件概念

插件（Extend）是 Vigo 框架的扩展机制，用于：

- **模块化**: 将功能封装为独立的插件模块
- **可插拔**: 支持动态加载和卸载插件
- **生命周期**: 提供初始化、运行、停止等生命周期钩子
- **依赖管理**: 自动处理插件间的依赖关系

### 1.2 插件 vs 门面

| 特性 | 插件（Extend） | 门面（Facade） |
| :--- | :--- | :--- |
| **用途** | 功能扩展 | 访问接口 |
| **生命周期** | Init/Run/Stop | Boot |
| **注册方式** | `extend.Register()` | `extend.RegisterFacade()` |
| **示例** | 支付插件、短信插件 | Redis 门面、DB 门面 |

## 二、插件接口

### 2.1 IExtend 接口

所有插件都必须实现 `IExtend` 接口：

```go
type IExtend interface {
    // Name 返回插件名称
    Name() string
    
    // Init 初始化插件
    Init() error
    
    // Run 运行插件
    Run() error
    
    // Stop 停止插件
    Stop() error
}
```

### 2.2 IFacade 接口

门面插件实现 `IFacade` 接口：

```go
type IFacade interface {
    // Name 返回门面名称
    Name() string
    
    // Boot 引导门面（应用启动时调用）
    Boot() error
    
    // GetAccessor 返回访问器
    GetAccessor() interface{}
}
```

### 3.2 注册插件

```go
// app/extend/init.go
package extend

import (
    "log"
    "vigo/framework/extend"
    "vigo/app/extend/payment"
    "vigo/app/extend/sms"
    "vigo/app/extend/storage"
)

// Init 初始化所有插件
func Init() {
    // 注册支付插件
    if err := extend.Register(&payment.PaymentPlugin{}); err != nil {
        log.Fatal("注册支付插件失败:", err)
    }
    
    // 注册短信插件
    if err := extend.Register(&sms.SmsPlugin{}); err != nil {
        log.Fatal("注册短信插件失败:", err)
    }
    
    // 注册存储插件
    if err := extend.Register(&storage.StoragePlugin{}); err != nil {
        log.Fatal("注册存储插件失败:", err)
    }
    
    // 初始化所有插件
    if err := extend.InitAll(); err != nil {
        log.Fatal("初始化插件失败:", err)
    }
    
    // 运行所有插件
    if err := extend.RunAll(); err != nil {
        log.Fatal("运行插件失败:", err)
    }
}
```

## 四、使用插件

### 4.1 获取插件实例

```go
// app/controller/order.go
package controller

import (
    "vigo/framework/mvc"
    "vigo/framework/extend"
)

// CreateOrder 创建订单
func CreateOrder(c *mvc.Context) {
    // 获取支付插件
    paymentPlugin := extend.GetExtend("payment")
    if paymentPlugin == nil {
        c.Json(500, map[string]interface{}{
            "code":    500,
            "message": "支付插件未加载",
        })
        return
    }
    
    // 类型断言
    payment := paymentPlugin.(*payment.PaymentPlugin)
    
    // 调用插件方法
    orderID := "order_123"
    amount := 99.99
    
    if err := payment.Pay(orderID, amount); err != nil {
        c.Json(500, map[string]interface{}{
            "code":    500,
            "message": "支付失败：" + err.Error(),
        })
        return
    }
    
    c.Json(200, map[string]interface{}{
        "code":    0,
        "message": "支付成功",
    })
}
```

### 4.2 创建插件门面

```go
// app/extend/payment/facade.go
package payment

import (
    "vigo/framework/extend"
)

// PaymentFacade 支付门面
type PaymentFacade struct {
    plugin *PaymentPlugin
}

// Name 返回门面名称
func (p *PaymentFacade) Name() string {
    return "payment"
}

// Boot 引导门面
func (p *PaymentFacade) Boot() error {
    // 获取插件实例
    p.plugin = extend.GetExtend("payment").(*PaymentPlugin)
    return nil
}

// GetAccessor 返回访问器
func (p *PaymentFacade) GetAccessor() interface{} {
    return p
}

// Pay 支付（委托给插件）
func (p *PaymentFacade) Pay(orderID string, amount float64) error {
    return p.plugin.Pay(orderID, amount)
}

// Refund 退款
func (p *PaymentFacade) Refund(orderID string) error {
    return p.plugin.Refund(orderID)
}

// Query 查询
func (p *PaymentFacade) Query(orderID string) (map[string]interface{}, error) {
    return p.plugin.Query(orderID)
}
```

### 4.3 注册门面

```go
// app/extend/init.go
func Init() {
    // 注册插件
    extend.Register(&payment.PaymentPlugin{})
    
    // 注册门面
    extend.RegisterFacade(&payment.PaymentFacade{})

    extend.RunAll()
```
```go
}
```

### 5.3 存储插件

```go
// app/extend/storage/storage.go
package storage

import (
    "io"
    "log"
    "mime/multipart"
    "os"
    "path/filepath"
)

// StoragePlugin 存储插件
type StoragePlugin struct {
    basePath string
}

// Name 返回插件名称
func (s *StoragePlugin) Name() string {
    return "storage"
}

// Init 初始化
func (s *StoragePlugin) Init() error {
    log.Println("[StoragePlugin] 初始化存储插件")
    s.basePath = "./storage"
    
    // 创建基础目录
    os.MkdirAll(s.basePath, 0755)
    
    return nil
}

// Run 运行
func (s *StoragePlugin) Run() error {
    log.Println("[StoragePlugin] 运行存储插件")
    return nil
}

// Stop 停止
func (s *StoragePlugin) Stop() error {
    log.Println("[StoragePlugin] 停止存储插件")
    return nil
}

// SaveFile 保存文件
func (s *StoragePlugin) SaveFile(file *multipart.FileHeader, path string) (string, error) {
    src, err := file.Open()
    if err != nil {
        return "", err
    }
    defer src.Close()
    
    // 完整路径
    fullPath := filepath.Join(s.basePath, path)
    
    // 创建目录
    dir := filepath.Dir(fullPath)
    os.MkdirAll(dir, 0755)
    
    // 创建文件
    dst, err := os.Create(fullPath)
    if err != nil {
        return "", err
    }
    defer dst.Close()
    
    // 复制内容
    _, err = io.Copy(dst, src)
    if err != nil {
        return "", err
    }
    
    log.Printf("[StoragePlugin] 保存文件：%s", fullPath)
    
    return fullPath, nil
}

// GetFile 获取文件
func (s *StoragePlugin) GetFile(path string) ([]byte, error) {
    fullPath := filepath.Join(s.basePath, path)
    return os.ReadFile(fullPath)
}

// DeleteFile 删除文件
func (s *StoragePlugin) DeleteFile(path string) error {
    fullPath := filepath.Join(s.basePath, path)
    return os.Remove(fullPath)
}

// Exists 检查文件是否存在
func (s *StoragePlugin) Exists(path string) bool {
    fullPath := filepath.Join(s.basePath, path)
    _, err := os.Stat(fullPath)
    return err == nil
}
```

## 六、插件管理器

### 6.1 手动管理插件

```go
import (
    "vigo/framework/extend"
    "vigo/app/extend/payment"
)

// 注册插件
plugin := &payment.PaymentPlugin{}
err := extend.Register(plugin)
if err != nil {
    log.Fatal("注册失败:", err)
}

// 初始化插件
err = extend.InitExtend("payment")
if err != nil {
    log.Fatal("初始化失败:", err)
}

// 运行插件
err = extend.RunExtend("payment")
if err != nil {
    log.Fatal("运行失败:", err)
}

// 获取插件
plugin := extend.GetExtend("payment")

// 停止插件
err = extend.StopExtend("payment")

// 卸载插件
err = extend.Unregister("payment")
```

### 6.2 批量管理

```go
// 初始化所有插件
err := extend.InitAll()

// 运行所有插件
err := extend.RunAll()

// 停止所有插件
err := extend.StopAll()
```

```go
// app/extend/order/order.go
package order

import (
    "vigo/framework/extend"

// OrderPlugin 订单插件
type OrderPlugin struct {
    payment *payment.PaymentPlugin

func (o *OrderPlugin) Name() string {
    return "order"

// Init 初始化
func (o *OrderPlugin) Init() error {
    // 获取依赖的插件
    paymentExtend := extend.GetExtend("payment")
    if paymentExtend == nil {
        return fmt.Errorf("缺少依赖插件：payment")
    }
    
    o.payment = paymentExtend.(*payment.PaymentPlugin)
    
    return nil
}

// Run 运行
func (o *OrderPlugin) Run() error {
    return nil
}

// Stop 停止
func (o *OrderPlugin) Stop() error {
    return nil
}

// CreateOrder 创建订单
func (o *OrderPlugin) CreateOrder(orderID string, amount float64) error {
    // 使用支付插件
    return o.payment.Pay(orderID, amount)
}
```

### 7.2 依赖顺序

```go
// main.go
func main() {
    // 1. 注册插件（按依赖顺序）
    extend.Register(&payment.PaymentPlugin{})  // 先注册被依赖的
    extend.Register(&order.OrderPlugin{})      // 后注册依赖的
    
    // 2. 初始化（自动按依赖顺序）
    extend.InitAll()
    
    // 3. 运行
    extend.RunAll()
    
    // 4. 启动应用
    app.Run()
}
```
