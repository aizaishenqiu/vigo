# 路由使用详解

Vigo 框架提供强大的路由系统，支持多种 HTTP 方法、路由组、中间件、参数绑定等特性。

---

## 一、基础路由

### 1.1 定义路由

在 `app/route/route.go` 中定义路由：

```go
package route

import (
    "vigo/framework/mvc"
    "vigo/app/controller"
)

func Init() {
    r := mvc.NewRouter()
    
    // GET 请求
    r.GET("/", controller.Index)
    r.GET("/hello", controller.Hello)
    
    // POST 请求
    r.POST("/user/create", controller.CreateUser)
    
    // PUT 请求
    r.PUT("/user/update", controller.UpdateUser)
    
    // DELETE 请求
    r.DELETE("/user/delete", controller.DeleteUser)
    
    // PATCH 请求
    r.PATCH("/user/patch", controller.PatchUser)
    
    // OPTIONS 请求
    r.OPTIONS("/user/options", controller.OptionsUser)
    
    // 匹配任意 HTTP 方法
    r.Any("/user/any", controller.AnyMethod)
}
```

### 1.2 路由参数

#### 必需参数

```go
// 定义带参数的路由
r.GET("/user/:id", controller.GetUser)
r.GET("/post/:postId/comment/:commentId", controller.GetComment)

// 控制器中获取参数
func GetUser(c *mvc.Context) {
    // 方式 1: 获取字符串参数
    id := c.Param("id")
    
    // 方式 2: 获取 int64 参数
    id := c.ParamInt64("id")
    
    // 方式 3: 获取 int 参数
    id := c.ParamInt("id")
    
    c.Json(200, map[string]interface{}{
        "user_id": id,
    })
}
```

#### 可选参数

```go
// 定义可选参数（使用 ? 标记）
r.GET("/user/:id?", controller.GetUserOptional)

// 控制器中处理
func GetUserOptional(c *mvc.Context) {
    id := c.Param("id")
    if id == "" {
        // 参数不存在，使用默认逻辑
        c.Json(200, map[string]interface{}{
            "message": "获取所有用户",
        })
        return
    }
    
    // 参数存在，获取单个用户
    c.Json(200, map[string]interface{}{
        "user_id": id,
    })
}
```

### 1.3 查询参数

```go
// 定义路由（不需要特别声明查询参数）
r.GET("/users", controller.ListUsers)

// 控制器中获取查询参数
func ListUsers(c *mvc.Context) {
    // 获取单个查询参数
    page := c.Query("page", "1")  // 带默认值
    limit := c.Query("limit", "10")
    
    // 获取 int 类型查询参数
    page := c.QueryInt("page", 1)
    limit := c.QueryInt("limit", 10)
    
    // 获取所有查询参数
    queries := c.Queries()
    // queries = map[string]string{"page": "1", "limit": "10", "keyword": "test"}
    
    c.Json(200, map[string]interface{}{
        "page": page,
        "limit": limit,
        "all": queries,
    })
}

// 访问：/users?page=2&limit=20&keyword=test
```

### 1.4 POST 数据

```go
// 定义 POST 路由
r.POST("/user/create", controller.CreateUser)

// 控制器中获取 POST 数据
func CreateUser(c *mvc.Context) {
    // 方式 1: 获取单个表单字段
    name := c.PostForm("name")
    email := c.PostForm("email", "default@example.com")  // 带默认值
    
    // 方式 2: 获取 JSON 数据
    var data map[string]interface{}
    if err := c.BindJSON(&data); err != nil {
        c.Json(400, map[string]interface{}{
            "code":    400,
            "message": "参数错误",
        })
        return
    }
    
    // 方式 3: 绑定到结构体
    type CreateUserRequest struct {
        Name  string `json:"name"`
        Email string `json:"email"`
        Age   int    `json:"age"`
    }
    
    var req CreateUserRequest
    if err := c.BindJSON(&req); err != nil {
        c.Json(400, map[string]interface{}{
            "code":    400,
            "message": "参数错误",
        })
        return
    }
    
    c.Json(200, map[string]interface{}{
        "code": 0,
        "data": req,
    })
}
```

## 二、路由组

### 2.1 基础路由组

```go
func Init() {
    r := mvc.NewRouter()
    
    // 定义路由组
    api := r.Group("/api")
    {
        // /api/users
        api.GET("/users", controller.ListUsers)
        api.POST("/users", controller.CreateUser)
        
        // /api/posts
        api.GET("/posts", controller.ListPosts)
        api.POST("/posts", controller.CreatePost)
    }
    
    // 嵌套路由组
    admin := r.Group("/admin")
    {
        user := admin.Group("/user")
        {
            // /admin/user/list
            user.GET("/list", controller.AdminUserList)
            // /admin/user/create
            user.POST("/create", controller.AdminUserCreate)
        }
        
        post := admin.Group("/post")
        {
            // /admin/post/list
            post.GET("/list", controller.AdminPostList)
        }
    }
}
```

### 2.2 路由组中间件

```go
func Init() {
    r := mvc.NewRouter()
    
    // 带中间件的路由组
    api := r.Group("/api")
    api.Use(middleware.Auth())  // 认证中间件
    api.Use(middleware.CORS())  // 跨域中间件
    {
        // 这些路由都会经过 Auth 和 CORS 中间件
        api.GET("/users", controller.ListUsers)
        api.POST("/users", controller.CreateUser)
    }
    
    // 公开路由（不需要认证）
    public := r.Group("/public")
    {
        public.GET("/info", controller.GetInfo)
        public.GET("/health", controller.HealthCheck)
    }
}
```

## 三、中间件

### 3.1 定义中间件

```go
// app/middleware/auth.go
package middleware

import (
    "net/http"
    "vigo/framework/mvc"
)

// Auth 认证中间件
func Auth() mvc.Middleware {
    return func(c *mvc.Context) {
        // 获取 Token
        token := c.GetHeader("Authorization")
        
        // 验证 Token
        if token == "" {
            c.Json(http.StatusUnauthorized, map[string]interface{}{
                "code":    401,
                "message": "未授权",
            })
            c.Abort()  // 终止后续处理
            return
        }
        
        // 验证通过，继续处理
        c.Next()
    }
}

// CORS 跨域中间件
func CORS() mvc.Middleware {
    return func(c *mvc.Context) {
        // 设置跨域头
        c.SetHeader("Access-Control-Allow-Origin", "*")
        c.SetHeader("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS")
        c.SetHeader("Access-Control-Allow-Headers", "Content-Type, Authorization")
        
        // 处理预检请求
        if c.Request.Method == "OPTIONS" {
            c.AbortWithStatus(204)
            return
        }
        
        c.Next()
    }
}
```

### 3.2 注册中间件

```go
func Init() {
    r := mvc.NewRouter()
    
    // 全局中间件（对所有路由生效）
    r.Use(middleware.CORS())
    r.Use(middleware.Logger())
    
    // 路由组中间件
    api := r.Group("/api")
    api.Use(middleware.Auth())
    {
        api.GET("/users", controller.ListUsers)
    }
    
    // 单个路由中间件
    r.GET("/admin/dashboard", controller.Dashboard, middleware.Auth(), middleware.AdminCheck())
}
```

## 四、高级路由

### 4.1 路由命名

```go
func Init() {
    r := mvc.NewRouter()
    
    // 命名路由
    r.GET("/user/profile", controller.Profile).Name("user.profile")
    r.GET("/user/settings", controller.Settings).Name("user.settings")
}
```

### 4.2 路由匹配顺序

路由匹配遵循以下顺序：

1. **静态路由**（优先级最高）
2. **参数路由**
3. **通配符路由**（优先级最低）

### 4.3 通配符路由

```go
// 匹配 /user/ 后面的所有内容
r.GET("/user/*action", controller.UserAny)

// 控制器中获取通配符内容
func UserAny(c *mvc.Context) {
    action := c.Param("action")
    c.Json(200, map[string]interface{}{
        "action": action,
    })
}
```

## 五、错误处理

### 5.1 404 处理

```go
func Init() {
    r := mvc.NewRouter()
    
    // 定义正常路由
    r.GET("/user/:id", controller.GetUser)
    
    // 404 处理（必须放在最后）
    r.NotFound(func(c *mvc.Context) {
        c.Json(404, map[string]interface{}{
            "code":    404,
            "message": "页面不存在",
        })
    })
}
```

### 5.2 全局错误处理

```go
func Init() {
    r := mvc.NewRouter()
    
    // 设置全局错误处理器
    r.SetErrorHandler(func(c *mvc.Context, err error) {
        log.Error("路由错误:", err)
        c.Json(500, map[string]interface{}{
            "code":    500,
            "message": "服务器错误",
        })
    })
}
```

## 六、完整示例

### 6.1 路由配置文件

```go
// app/route/route.go
package route

import (
    "vigo/framework/mvc"
    "vigo/app/controller"
    "vigo/app/middleware"
)

func Init() {
    r := mvc.NewRouter()
    
    // 全局中间件
    r.Use(middleware.CORS())
    r.Use(middleware.Logger())
    
    // 首页
    r.GET("/", controller.Index)
    
    // 公开 API
    public := r.Group("/api/public")
    {
        public.GET("/info", controller.GetInfo)
        public.POST("/login", controller.Login)
        public.POST("/register", controller.Register)
    }
    
    // 需要认证的 API
    api := r.Group("/api")
    api.Use(middleware.Auth())
    {
        // 用户相关
        user := api.Group("/user")
        {
            user.GET("/profile", controller.GetProfile)
            user.PUT("/profile", controller.UpdateProfile)
            user.POST("/avatar", controller.UploadAvatar)
        }
        
        // 文章相关
        post := api.Group("/post")
        {
            post.GET("/", controller.ListPosts)
            post.POST("/", controller.CreatePost)
            post.GET("/:id", controller.GetPost)
            post.PUT("/:id", controller.UpdatePost)
            post.DELETE("/:id", controller.DeletePost)
        }
    }
    
    // 后台管理
    admin := r.Group("/admin")
    admin.Use(middleware.Auth(), middleware.AdminCheck())
    {
        admin.GET("/dashboard", controller.Dashboard)
        admin.GET("/users", controller.AdminUsers)
        admin.GET("/posts", controller.AdminPosts)
    }
    
    // 404 处理
    r.NotFound(func(c *mvc.Context) {
        c.Json(404, map[string]interface{}{
            "code":    404,
            "message": "页面不存在",
        })
    })
}
```

## 七、最佳实践

### 7.1 路由组织

- 按功能模块组织路由
- 使用路由组统一管理
- 合理命名路由，便于维护

- 全局中间件：跨域、日志等通用功能
- 路由组中间件：认证、权限等业务功能
- 单个路由中间件：特殊处理

### 7.3 参数验证

- 在控制器中验证参数
- 使用验证器进行复杂验证
- 返回友好的错误信息

### 7.4 错误处理

- 统一错误响应格式
- 记录详细错误日志
- 返回简洁的错误信息给用户
