# 模型（Model）操作

## 简介

模型是MVC架构中的数据处理层，负责数据库操作和业务逻辑。框架提供了 ThinkPHP/Laravel 风格的 ORM 操作。

## 模型使用方式

框架提供两种模型使用方式：

### 方式一：使用 Model 基类（推荐）

框架提供了功能丰富的 Model 基类，支持自动时间戳、软删除、事件回调、获取器/修改器等特性。

```go
import "vigo/framework/model"

// 创建模型实例
userModel := model.New("users")

// 配置模型
userModel.SetPk("id").
    SetAutoTimestamp(true).
    SetSoftDelete(true, "delete_time")

// 查询单条
user := userModel.Where("id", 1).Find()
fmt.Println(user.GetData()["name"])

// 查询多条
users, _ := userModel.Where("status", 1).Select()
for _, u := range users {
    fmt.Println(u.GetData()["name"])
}

// 创建记录
newUser := model.New("users")
newUser.Data(map[string]interface{}{
    "name":  "张三",
    "email": "zhangsan@example.com",
})
id, _ := newUser.Save()

// 更新记录
user := model.New("users").Find(1)
user.Data(map[string]interface{}{"name": "李四"})
affected, _ := user.Save()

// 删除记录（软删除）
user := model.New("users").Find(1)
affected, _ := user.Delete()
```

### 方式二：自定义模型结构体

适合需要封装复杂业务逻辑的场景。

```go
package model

import (
    "time"
    "vigo/framework/db"
)

// UserModel 用户模型
type UserModel struct {
    TableName string
}

// NewUserModel 创建用户模型实例
func NewUserModel() *UserModel {
    return &UserModel{
        TableName: "users",
    }
}
```

## 查询操作

### FindByID - 根据ID查询

```go
func (m *UserModel) FindByID(id int64) (map[string]interface{}, error) {
    return db.Table(m.TableName).
        Where("id", id).
        WhereNull("delete_time").
        Find()
}
```

### FindByEmail - 根据邮箱查询

```go
func (m *UserModel) FindByEmail(email string) (map[string]interface{}, error) {
    return db.Table(m.TableName).
        Where("email", email).
        WhereNull("delete_time").
        Find()
}
```

### GetList - 获取列表

```go
func (m *UserModel) GetList(page, pageSize int) ([]map[string]interface{}, int64, error) {
    query := db.Table(m.TableName).WhereNull("delete_time")

    // 统计总数
    total, err := query.Count()
    if err != nil {
        return nil, 0, err
    }

    // 查询列表
    list, err := db.Table(m.TableName).
        WhereNull("delete_time").
        Order("id DESC").
        Page(page, pageSize).
        Select()

    return list, total, err
}
```

### GetByStatus - 根据状态查询

```go
func (m *UserModel) GetByStatus(status int) ([]map[string]interface{}, error) {
    return db.Table(m.TableName).
        Where("status", status).
        WhereNull("delete_time").
        Order("create_time DESC").
        Select()
}
```

## 事务操作

### 使用事务

```go
func (m *UserModel) TransferPoints(fromID, toID int64, points int) error {
    return db.Transaction(func(q *db.Query) error {
        // 扣减积分
        affected, err := q.Table(m.TableName).
            Where("id", fromID).
            Where("points", ">=", points).
            Dec("points", points)
        if err != nil {
            return err
        }
        if affected == 0 {
            return fmt.Errorf("积分不足")
        }

        // 增加积分
        _, err = q.Table(m.TableName).
            Where("id", toID).
            Inc("points", points)
        if err != nil {
            return err
        }

        // 记录日志
        _, err = q.Table("point_logs").Insert(map[string]interface{}{
            "from_id": fromID,
            "to_id":   toID,
            "points":  points,
            "create_time": time.Now().Format("2006-01-02 15:04:05"),
        })

        return err
    })
}
```

## 完整模型示例

```go
package model

import (
    "fmt"
    "time"
    "vigo/framework/db"
)

// UserModel 用户模型
type UserModel struct {
    TableName string
}

func NewUserModel() *UserModel {
    return &UserModel{TableName: "users"}
}

// FindByID 根据ID查询
func (m *UserModel) FindByID(id int64) (map[string]interface{}, error) {
    return db.Table(m.TableName).
        Where("id", id).
        WhereNull("delete_time").
        Find()
}

// FindByEmail 根据邮箱查询
func (m *UserModel) FindByEmail(email string) (map[string]interface{}, error) {
    return db.Table(m.TableName).
        Where("email", email).
        WhereNull("delete_time").
        Find()
}

// GetList 获取列表
func (m *UserModel) GetList(page, pageSize int, filters map[string]interface{}) ([]map[string]interface{}, int64, error) {
    query := db.Table(m.TableName).WhereNull("delete_time")

    // 应用过滤条件
    if status, ok := filters["status"]; ok {
        query = query.Where("status", status)
    }
    if city, ok := filters["city"]; ok {
        query = query.Where("city", city)
    }
    if keyword, ok := filters["keyword"]; ok {
        query = query.WhereLike("name", keyword.(string))
    }

    // 统计总数
    total, err := query.Count()
    if err != nil {
        return nil, 0, err
    }

    // 查询列表
    list, err := query.
        Order("id DESC").
        Page(page, pageSize).
        Select()

    return list, total, err
}

// Create 创建用户
func (m *UserModel) Create(data map[string]interface{}) (int64, error) {
    now := time.Now().Format("2006-01-02 15:04:05")
    data["create_time"] = now
    data["update_time"] = now

    return db.Table(m.TableName).InsertGetId(data)
}

// Update 更新用户
func (m *UserModel) Update(id int64, data map[string]interface{}) (int64, error) {
    data["update_time"] = time.Now().Format("2006-01-02 15:04:05")

    return db.Table(m.TableName).
        Where("id", id).
        Update(data)
}

// Delete 软删除
func (m *UserModel) Delete(id int64) (int64, error) {
    return db.Table(m.TableName).
        Where("id", id).
        Update(map[string]interface{}{
            "delete_time": time.Now().Format("2006-01-02 15:04:05"),
        })
}

// Count 统计数量
func (m *UserModel) Count() (int64, error) {
    return db.Table(m.TableName).
        WhereNull("delete_time").
        Count()
}

// Exists 检查邮箱是否存在
func (m *UserModel) Exists(email string, excludeID int64) (bool, error) {
    query := db.Table(m.TableName).
        Where("email", email).
        WhereNull("delete_time")

    if excludeID > 0 {
        query = query.Where("id", "<>", excludeID)
    }

    return query.Exists()
}
```

## 注意事项

1. 模型应封装数据访问逻辑，不暴露数据库操作细节
2. 软删除应在查询时自动过滤已删除记录
3. 复杂业务逻辑使用事务保证数据一致性
4. 关联查询注意N+1问题，合理使用JOIN
5. 时间字段统一使用格式化字符串

## Model 基类特性

### 自动时间戳

```go
// 启用自动时间戳
userModel := model.New("users").SetAutoTimestamp(true)

// 创建记录时自动填充 create_time 和 update_time
userModel.Data(map[string]interface{}{"name": "张三"}).Save()

// 更新记录时自动更新 update_time
userModel.Find(1).Data(map[string]interface{}{"name": "李四"}).Save()
```

### 软删除

```go
// 启用软删除
userModel := model.New("users").SetSoftDelete(true, "delete_time")

// 删除时自动设置 delete_time
userModel.Find(1).Delete()

// 查询时自动过滤已删除记录
users, _ := userModel.Select()

// 包含已删除记录
users, _ := userModel.WithTrashed().Select()

// 仅查询已删除记录
users, _ := userModel.OnlyTrashed().Select()
```

### 模型事件

```go
userModel := model.New("users")

// 注册事件回调
userModel.BeforeInsert(func(m *model.Model) error {
    // 插入前验证
    if m.GetData()["email"] == "" {
        return fmt.Errorf("邮箱不能为空")
    }
    return nil
})

userModel.AfterInsert(func(m *model.Model) {
    // 插入后发送通知
    fmt.Println("用户创建成功:", m.GetData()["id"])
})

userModel.BeforeUpdate(func(m *model.Model) error {
    // 更新前验证
    return nil
})

userModel.AfterUpdate(func(m *model.Model) {
    // 更新后处理
})

userModel.BeforeDelete(func(m *model.Model) error {
    // 删除前验证
    return nil
})

userModel.AfterDelete(func(m *model.Model) {
    // 删除后清理
})
```

### 获取器和修改器

框架通过方法名约定实现获取器和修改器，无需额外配置。

```go
// 自定义模型
type UserModel struct {
    model.Model
}

// 获取器 - 方法名格式: Get{字段名}Attr
// 读取 status 字段时自动调用
func (m *UserModel) GetStatusTextAttr() interface{} {
    status, _ := m.GetData()["status"].(int)
    switch status {
    case 1:
        return "正常"
    case 0:
        return "禁用"
    default:
        return "未知"
    }
}

// 修改器 - 方法名格式: Set{字段名}Attr
// 写入 password 字段时自动调用
func (m *UserModel) SetPasswordAttr(value interface{}) interface{} {
    // 自动加密密码
    return hashPassword(value.(string))
}

// 使用示例
user := &UserModel{}
user.Model = *model.New("users")

// 设置密码时自动加密
user.SetAttr("password", "123456")

// 获取状态文本
user.Find(1)
fmt.Println(user.GetAttr("status_text")) // 输出: 正常
```

### 关联查询

```go
user := userModel.Find(1)

// 一对一关联 - 用户资料
profile := user.HasOne("user_profiles", "user_id", "id")

// 一对多关联 - 用户订单
orders := user.HasMany("orders", "user_id", "id")

// 反向一对一 - 订单所属用户
order := model.New("orders").Find(1)
user := order.BelongsTo("users", "user_id", "id")

// 多对多关联 - 用户角色
roles := user.BelongsToMany("roles", "user_roles", "user_id", "role_id")

// 远程一对一 - 国家通过用户资料
country := user.HasOneThrough("countries", "user_profiles", "id", "user_id", "country_id", "country_id")

// 远程一对多 - 国家所有用户订单
orders := user.HasManyThrough("orders", "user_profiles", "id", "user_id", "user_id", "user_id")
```
