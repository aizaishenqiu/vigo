# 控制器（Controller）操作

## 简介
控制器是MVC架构的协调者，负责接收HTTP请求、调用模型处理数据、返回响应。框架提供了完整的控制器支持。

## 控制器定义

### 基础控制器
```go
package controller

import "vigo/framework/mvc"

// Controller 基础控制器
type Controller struct {
    mvc.Controller
}

// Init 初始化控制器
func (c *Controller) Init(ctx *mvc.Context) {
    c.Ctx = ctx
}
```

### 用户控制器示例
```go
package controller

import (
    "myapp/model"
    "vigo/framework/mvc"
)

// UserController 用户控制器
type UserController struct {
    Controller
}

// List 用户列表
func (uc *UserController) List(ctx *mvc.Context) {
    page := ctx.InputInt("page", 1)
    pageSize := ctx.InputInt("page_size", 10)
    
    // 调用模型获取数据
    userModel := &model.UserModel{}
    users, total, err := userModel.GetList(page, pageSize)
    if err != nil {
        ctx.Error(500, "获取用户列表失败")
        return
    }
    
    // 返回成功响应
    ctx.Success(map[string]interface{}{
        "list":  users,
        "total": total,
        "page":  page,
        "page_size": pageSize,
    })
}

// Detail 用户详情
func (uc *UserController) Detail(ctx *mvc.Context) {
    id := ctx.Param("id")
    if id == "" {
        ctx.Error(400, "缺少用户ID")
        return
    }
    
    userModel := &model.UserModel{}
    user, err := userModel.FindByID(id)
    if err != nil {
        ctx.Error(500, "获取用户信息失败")
        return
    }
    
    if user == nil {
        ctx.Error(404, "用户不存在")
        return
    }
    
    ctx.Success(user)
}

// Create 创建用户
func (uc *UserController) Create(ctx *mvc.Context) {
    var req struct {
        Name   string `json:"name"`
        Email  string `json:"email"`
        Phone  string `json:"phone"`
    }
    
    if err := ctx.BindJSON(&req); err != nil {
        ctx.Error(400, "请求数据格式错误")
        return
    }
    
    // 验证数据
    if req.Name == "" {
        ctx.Error(400, "用户名不能为空")
        return
    }
    if req.Email == "" {
        ctx.Error(400, "邮箱不能为空")
        return
    }
    
    // 调用模型创建用户
    userModel := &model.UserModel{}
    id, err := userModel.Create(map[string]interface{}{
        "name":  req.Name,
        "email": req.Email,
        "phone": req.Phone,
    })
    if err != nil {
        ctx.Error(500, "创建用户失败")
        return
    }
    
    ctx.Success(map[string]interface{}{"id": id})
}

// Update 更新用户
func (uc *UserController) Update(ctx *mvc.Context) {
    id := ctx.Param("id")
    if id == "" {
        ctx.Error(400, "缺少用户ID")
        return
    }
    
    var req struct {
        Name  string `json:"name"`
        Email string `json:"email"`
        Phone string `json:"phone"`
    }
    
    if err := ctx.BindJSON(&req); err != nil {
        ctx.Error(400, "请求数据格式错误")
        return
    }
    
    userModel := &model.UserModel{}
    affected, err := userModel.Update(id, map[string]interface{}{
        "name":  req.Name,
        "email": req.Email,
        "phone": req.Phone,
    })
    if err != nil {
        ctx.Error(500, "更新用户失败")
        return
    }
    
    ctx.Success(map[string]interface{}{"affected": affected})
}

// Delete 删除用户
func (uc *UserController) Delete(ctx *mvc.Context) {
    id := ctx.Param("id")
    if id == "" {
        ctx.Error(400, "缺少用户ID")
        return
    }
    
    userModel := &model.UserModel{}
    affected, err := userModel.Delete(id)
    if err != nil {
        ctx.Error(500, "删除用户失败")
        return
    }
    
    ctx.Success(map[string]interface{}{"affected": affected})
}
```

## Context请求方法

### 获取请求参数
```go
// 获取表单/查询参数
name := ctx.Input("name")
page := ctx.InputInt("page", 1)           // 带默认值
price := ctx.InputFloat("price", 0.0)     // 浮点型
active := ctx.InputBool("active")         // 布尔型
name := ctx.InputDefault("name", "guest") // 带默认值

// 获取URL查询参数
keyword := ctx.Query("keyword")
page := ctx.QueryDefault("page", "1")

// 获取路由参数
id := ctx.Param("id")

// 获取Cookie
token, err := ctx.Cookie("token")

// 获取请求头
auth := ctx.GetHeader("Authorization")
contentType := ctx.GetHeader("Content-Type")

// 获取客户端IP
ip := ctx.GetClientIP()

// 绑定JSON到结构体
var user struct {
    Name  string `json:"name"`
    Email string `json:"email"`
}
err := ctx.BindJSON(&user)

// 获取上传文件
file, err := ctx.FormFile("avatar")
```

### 请求判断方法
```go
// 获取请求路径
path := ctx.Path()

// 判断是否AJAX请求
if ctx.IsAjax() {
    // AJAX请求处理
}

// 获取Content-Type
ct := ctx.ContentType()

// 获取请求方法
method := ctx.Request.Method
```

## Context响应方法

### JSON响应
```go
// 成功响应
ctx.Success(data)
// 返回: {"code":0,"msg":"success","data":{...}}

// 错误响应
ctx.Error(400, "参数错误")
// 返回: {"code":400,"msg":"参数错误","data":null}

// 自定义JSON响应
ctx.Json(200, map[string]interface{}{
    "code": 0,
    "data": users,
})

// 结构化失败响应
ctx.Fail(200, 10001, "用户不存在", nil)
```

### HTML响应
```go
// 渲染模板
ctx.HTML(200, "user/detail.html", map[string]interface{}{
    "Title": "用户详情",
    "User":  user,
})
```

### 文本响应
```go
// 纯文本响应
ctx.String(200, "Hello %s", name)
```

### 重定向
```go
// 临时重定向
ctx.Redirect(302, "/login")

// 永久重定向
ctx.Redirect(301, "https://example.com")
```

### 响应头设置
```go
// 设置响应头
ctx.SetHeader("X-Custom-Header", "value")
ctx.SetHeader("Cache-Control", "no-cache")
```

## 中间件

### 定义中间件
```go
package middleware

import "vigo/framework/mvc"

// Auth 认证中间件
func Auth(ctx *mvc.Context) {
    token := ctx.GetHeader("Authorization")
    if token == "" {
        ctx.Error(401, "未授权访问")
        ctx.Abort()
        return
    }
    
    // 验证token
    userID, err := validateToken(token)
    if err != nil {
        ctx.Error(401, "无效的token")
        ctx.Abort()
        return
    }
    
    // 存储用户ID到上下文
    ctx.Set("user_id", userID)
    ctx.Next()
}

// Logger 日志中间件
func Logger(ctx *mvc.Context) {
    start := time.Now()
    
    ctx.Next()
    
    elapsed := time.Since(start)
    log.Printf("[%s] %s %d %v",
        ctx.Request.Method,
        ctx.Path(),
        ctx.Status(),
        elapsed,
    )
}

// CORS 跨域中间件
func CORS(ctx *mvc.Context) {
    ctx.SetHeader("Access-Control-Allow-Origin", "*")
    ctx.SetHeader("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE,OPTIONS")
    ctx.SetHeader("Access-Control-Allow-Headers", "Content-Type,Authorization")
    
    if ctx.Request.Method == "OPTIONS" {
        ctx.Abort()
        return
    }
    
    ctx.Next()
}
```

### 注册中间件
```go
// 全局中间件
router.Use(middleware.Logger, middleware.CORS)

// 路由分组中间件
api := router.Group("/api", middleware.Auth)

// 单个路由中间件
router.GET("/profile", middleware.Auth, userController.Profile)
```

## 路由注册

### 基本路由
```go
// 创建路由器
router := mvc.NewRouter()

// 基本路由
router.GET("/users", userController.List)
router.POST("/users", userController.Create)
router.PUT("/users/:id", userController.Update)
router.DELETE("/users/:id", userController.Delete)

// 所有HTTP方法
router.Any("/api", handler)
```

### 路由分组
```go
// API分组
api := router.Group("/api")
api.GET("/users", userController.List)
api.POST("/users", userController.Create)

// 嵌套分组
v1 := router.Group("/api/v1")
{
    users := v1.Group("/users")
    users.GET("", userController.List)
    users.POST("", userController.Create)
    users.GET("/:id", userController.Detail)
}

// 分组中间件
admin := router.Group("/admin", middleware.Auth, middleware.Admin)
admin.GET("/users", adminController.Users)
```

### 动态路由
```go
// 参数路由
router.GET("/users/:id", userController.Detail)
router.GET("/posts/:id/comments/:comment_id", postController.Comment)

// 通配符路由
router.GET("/files/*filepath", fileController.Download)
```

## 完整示例

### 页面控制器
```go
package controller

import "vigo/framework/mvc"

// PageController 页面控制器
type PageController struct {
    Controller
}

// Home 首页
func (pc *PageController) Home(ctx *mvc.Context) {
    ctx.HTML(200, "home.html", map[string]interface{}{
        "Title": "首页",
    })
}

// UserList 用户列表页
func (pc *PageController) UserList(ctx *mvc.Context) {
    page := ctx.InputInt("page", 1)
    
    userModel := &model.UserModel{}
    users, total, _ := userModel.GetList(page, 10)
    
    ctx.HTML(200, "user/list.html", map[string]interface{}{
        "Title": "用户列表",
        "Users": users,
        "Total": total,
        "Page":  page,
    })
}

// UserDetail 用户详情页
func (pc *PageController) UserDetail(ctx *mvc.Context) {
    id := ctx.Param("id")
    
    userModel := &model.UserModel{}
    user, _ := userModel.FindByID(id)
    
    ctx.HTML(200, "user/detail.html", map[string]interface{}{
        "Title": "用户详情",
        "User":  user,
    })
}
```

## 注意事项
1. 控制器保持轻量，复杂业务逻辑放在模型层
2. 使用Context提供的方法处理请求和响应
3. 统一使用Success和Error方法返回API响应
4. 中间件中使用Abort中止请求处理
5. 路由参数使用`:id`格式，通配符使用`*filepath`
