# ORM 增强使用指南

Vigo 框架提供强大的 ORM 增强功能，包括软删除、JSON 查询、自动时间戳、获取器修改器等高级特性，对齐 ThinkPHP 8.1.4。

## 一、软删除

### 1.1 启用软删除

```go
package model

type Article struct {
    ID        int       `json:"id"`
    Title     string    `json:"title"`
    Content   string    `json:"content"`
    DeletedAt time.Time `json:"deleted_at,omitempty"`  // 软删除字段
}

// TableName 指定表名
func (a *Article) TableName() string {
    return "articles"
}
```

**数据库表结构**:
```sql
CREATE TABLE articles (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    content TEXT,
    deleted_at DATETIME DEFAULT NULL,  -- 软删除字段
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
```

---

### 1.2 软删除操作

```go
// 创建模型
article := model.New("articles")

// 软删除（不会真正删除数据）
err := article.Where("id", 1).Delete()
// SQL: UPDATE articles SET deleted_at = NOW() WHERE id = 1

// 强制删除（真正删除数据）
err = article.Where("id", 1).ForceDeleteByID(1)
// SQL: DELETE FROM articles WHERE id = 1

// 恢复已删除的数据
err = article.Where("id", 1).RestoreByID(1)
// SQL: UPDATE articles SET deleted_at = NULL WHERE id = 1

// 检查是否已删除
if article.Trashed() {
    fmt.Println("文章已删除")
}
```

---

### 1.3 查询软删除数据

```go
article := model.New("articles")

// 只查询未删除的数据（默认行为）
data := article.Where("status", 1).Select()

// 只查询已删除的数据
data = article.OnlyTrashed().Select()
// SQL: SELECT * FROM articles WHERE deleted_at IS NOT NULL AND status = 1

// 查询所有数据（包括已删除）
data = article.WithTrashed().Select()
// SQL: SELECT * FROM articles WHERE status = 1

// 查询时排除已删除
data = article.WithoutTrashed().Select()
// SQL: SELECT * FROM articles WHERE deleted_at IS NULL AND status = 1
```

---

### 1.4 批量操作

```go
// 批量软删除
article.Where("category_id", 5).Delete()
// SQL: UPDATE articles SET deleted_at = NOW() WHERE category_id = 5

// 批量恢复
article.Where("category_id", 5).Restore()
// SQL: UPDATE articles SET deleted_at = NULL WHERE category_id = 5

// 批量强制删除
article.Where("category_id", 5).ForceDelete()
// SQL: DELETE FROM articles WHERE category_id = 5
```

## 二、JSON 查询

### 2.1 JSON 字段查询

```go
// 表结构包含 JSON 字段
type User struct {
    ID       int                    `json:"id"`
    Name     string                 `json:"name"`
    Options  map[string]interface{} `json:"options"`  // JSON 字段
}
```

**数据库表结构**:
```sql
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    options JSON,  -- JSON 字段
    created_at DATETIME
);

-- 示例数据
INSERT INTO users (name, options) VALUES 
('张三', '{"city": "Beijing", "age": 25, "vip": true}'),
('李四', '{"city": "Shanghai", "age": 30, "vip": false}');
```

---

### 2.2 WhereJSON 查询

```go
user := model.New("users")

// 查询 JSON 字段等于指定值
data := user.WhereJSON("options", "city", "Beijing").Select()
// SQL: SELECT * FROM users WHERE options->>'$.city' = 'Beijing'

// 查询 JSON 字段等于数字
data = user.WhereJSON("options", "age", 25).Select()
// SQL: SELECT * FROM users WHERE options->>'$.age' = 25

// 查询 JSON 字段等于布尔值
data = user.WhereJSON("options", "vip", true).Select()
// SQL: SELECT * FROM users WHERE options->>'$.vip' = true
```

---

### 2.3 WhereJSONIn 查询

```go
user := model.New("users")

// 查询 JSON 字段在指定范围内
cities := []interface{}{"Beijing", "Shanghai", "Guangzhou"}
data := user.WhereJSONIn("options", "city", cities).Select()
// SQL: SELECT * FROM users WHERE options->>'$.city' IN ('Beijing', 'Shanghai', 'Guangzhou')
```

---

### 2.4 WhereJSONLike 查询

```go
user := model.New("users")

// 模糊查询 JSON 字段
data := user.WhereJSONLike("options", "city", "Bei%").Select()
// SQL: SELECT * FROM users WHERE options->>'$.city' LIKE 'Bei%'
```

---

### 2.5 WhereJSONContains 查询

```go
user := model.New("users")

// 查询 JSON 数组包含指定值
data := user.WhereJSONContains("options", "tags", "vip").Select()
// SQL: SELECT * FROM users WHERE JSON_CONTAINS(options->'$.tags', '"vip"')
```

---

### 2.6 获取 JSON 字段

```go
user := model.New("users")
user.Where("id", 1).First()

// 获取整个 JSON 字段
options, err := user.GetJSON("options")
if err != nil {
    log.Printf("获取 JSON 失败：%v", err)
}
fmt.Printf("Options: %+v\n", options)

// 获取 JSON 字段中的特定值
city := options["city"]
age := options["age"]
```

---

### 2.7 设置 JSON 字段

```go
user := model.New("users")

// 设置 JSON 字段
jsonData := map[string]interface{}{
    "city": "Beijing",
    "age":  25,
    "vip":  true,
}

err := user.Where("id", 1).SetJSON("options", jsonData)
// SQL: UPDATE users SET options = ? WHERE id = 1

// 更新 JSON 字段中的特定值
err = user.Where("id", 1).SetJSONField("options", "city", "Shanghai")
// SQL: UPDATE users SET JSON_SET(options, '$.city', ?) WHERE id = 1
```

## 三、自动时间戳

### 3.1 启用自动时间戳

```go
type Article struct {
    ID        int       `json:"id"`
    Title     string    `json:"title"`
    CreatedAt time.Time `json:"created_at"`
    UpdatedAt time.Time `json:"updated_at"`
}

// 自动管理时间戳
func (a *Article) BeforeCreate() {
    a.CreatedAt = time.Now()
    a.UpdatedAt = time.Now()
}

func (a *Article) BeforeUpdate() {
    a.UpdatedAt = time.Now()
}
```

---

### 3.2 使用时间戳 Trait

```go
package model

// HasTimestamps 自动时间戳 Trait
type HasTimestamps struct {
    CreatedAt time.Time `json:"created_at"`
    UpdatedAt time.Time `json:"updated_at"`
}

// BeforeCreate 创建前回调
func (h *HasTimestamps) BeforeCreate() {
    h.CreatedAt = time.Now()
    h.UpdatedAt = time.Now()
}

// BeforeUpdate 更新前回调
func (h *HasTimestamps) BeforeUpdate() {
    h.UpdatedAt = time.Now()
}

// 在模型中嵌入
type Article struct {
    HasTimestamps  // 嵌入时间戳 Trait
    ID        int   `json:"id"`
    Title     string `json:"title"`
}
```

---

### 3.3 自定义时间戳字段

```go
type Article struct {
    ID         int       `json:"id"`
    Title      string    `json:"title"`
    CreateTime time.Time `json:"create_time"`
    UpdateTime time.Time `json:"update_time"`
}

func (a *Article) BeforeCreate() {
    a.CreateTime = time.Now()
    a.UpdateTime = time.Now()
}

func (a *Article) BeforeUpdate() {
    a.UpdateTime = time.Now()

```

## 四、获取器和修改器

### 4.1 获取器（Accessor）

```go
type User struct {
    ID       int    `json:"id"`
    Email    string `json:"email"`
    Password string `json:"-"`  // 不输出
    Status   int    `json:"status"`
}

// GetEmailAttribute 获取器：将邮箱转为小写
func (u *User) GetEmailAttribute(value string) string {
    return strings.ToLower(value)
}

// GetStatusTextAttribute 获取器：返回状态文本
func (u *User) GetStatusTextAttribute() string {
    statusMap := map[int]string{
        0: "禁用",
        1: "正常",
        2: "锁定",
    }
    return statusMap[u.Status]
}

// 使用
user := &User{Email: "TEST@Example.com", Status: 1}
fmt.Println(user.Email)       // test@example.com
fmt.Println(user.StatusText)  // 正常
```

---

### 4.2 修改器（Mutator）

```go
type User struct {
    ID       int    `json:"id"`
    Email    string `json:"email"`
    Password string `json:"password"`
}

// SetEmailAttribute 修改器：保存时转为小写
func (u *User) SetEmailAttribute(value string) {
    u.Email = strings.ToLower(value)
}

// SetPasswordAttribute 修改器：保存时加密
func (u *User) SetPasswordAttribute(value string) {
    u.Password = bcrypt.HashPassword(value)
}

// 使用
user := &User{}
user.Email = "TEST@EXAMPLE.COM"
user.Password = "123456"

// 保存到数据库
db.Save(user)
// 数据库存储：email="test@example.com", password="$2a$10$..."
```

---

### 4.3 组合使用

```go
type Article struct {
    ID        int       `json:"id"`
    Title     string    `json:"title"`
    Content   string    `json:"content"`
    CreatedAt time.Time `json:"created_at"`
}

// 获取器：格式化标题
func (a *Article) GetTitleAttribute(value string) string {
    return strings.TrimSpace(value)
}

// 获取器：内容摘要
func (a *Article) GetSummaryAttribute() string {
    if len(a.Content) > 200 {
        return a.Content[:200] + "..."
    }
    return a.Content
}

// 修改器：保存前清理内容
func (a *Article) SetContentAttribute(value string) {
    a.Content = strings.TrimSpace(value)
}

// 使用
article := &Article{Title: "  Hello World  ", Content: "Some content..."}
fmt.Println(article.Title)    // "Hello World"
fmt.Println(article.Summary)  // 前 200 字符
```

## 五、只读字段

### 5.1 设置只读字段

```go
type Article struct {
    ID        int       `json:"id"`
    Title     string    `json:"title"`
    Content   string    `json:"content"`
    ViewCount int       `json:"view_count"`
    CreatedAt time.Time `json:"created_at"`
}

// Readonly 设置只读字段
func (a *Article) Readonly() []string {
    return []string{"id", "view_count", "created_at"}
}

// 使用
article := &Article{ID: 1, Title: "New Title", ViewCount: 100}

// 更新时，只读字段不会被更新
db.Model(article).Where("id", 1).Update(map[string]interface{}{
    "title":     "Updated Title",
    "view_count": 200,  // 不会被更新
})
// SQL: UPDATE articles SET title = ? WHERE id = 1
```

## 六、脏数据检测

### 6.1 检测字段变化

```go
type User struct {
    ID    int    `json:"id"`
    Name  string `json:"name"`
    Email string `json:"email"`
    
    original map[string]interface{}  // 原始数据
}

// 加载原始数据
func (u *User) SetOriginal(data map[string]interface{}) {
    u.original = data
}

// IsDirty 检测是否被修改
func (u *User) IsDirty(field string) bool {
    if u.original == nil {
        return false
    }
    
    currentValue := reflect.ValueOf(u).Elem().FieldByName(field).Interface()
    originalValue := u.original[field]
    
    return !reflect.DeepEqual(currentValue, originalValue)
}

// GetDirty 获取所有修改的字段
func (u *User) GetDirty() map[string]interface{} {
    dirty := make(map[string]interface{})
    
    if u.IsDirty("name") {
        dirty["name"] = u.Name
    }
    if u.IsDirty("email") {
        dirty["email"] = u.Email
    }
    
    return dirty
}

// 使用
user := &User{Name: "New Name", Email: "old@example.com"}
user.SetOriginal(map[string]interface{}{
    "name":  "Old Name",
    "email": "old@example.com",
})

fmt.Println(user.IsDirty("name"))   // true
fmt.Println(user.IsDirty("email"))  // false
fmt.Println(user.GetDirty())        // map[name:New Name]
```

## 七、完整示例

### 7.1 用户模型

```go
package model

import (
    "time"
    "vigo/framework/model"
)

type User struct {
    model.Model  // 嵌入基础模型
    
    ID          int       `json:"id"`
    Username    string    `json:"username"`
    Email       string    `json:"email"`
    Password    string    `json:"-"`
    Status      int       `json:"status"`
    Options     JSONField `json:"options"`  // JSON 字段
    DeletedAt   time.Time `json:"deleted_at,omitempty"`
    CreatedAt   time.Time `json:"created_at"`
    UpdatedAt   time.Time `json:"updated_at"`
}

// TableName 指定表名
func (u *User) TableName() string {
    return "users"
}

// 获取器
func (u *User) GetEmailAttribute(value string) string {
    return strings.ToLower(value)
}

func (u *User) GetStatusTextAttribute() string {
    statusMap := map[int]string{
        0: "禁用",
        1: "正常",
        2: "锁定",
    }
    return statusMap[u.Status]
}

// 修改器
func (u *User) SetEmailAttribute(value string) {
    u.Email = strings.ToLower(value)
}

func (u *User) SetPasswordAttribute(value string) {
    u.Password = bcrypt.HashPassword(value)
}

// 只读字段
func (u *User) Readonly() []string {
    return []string{"id", "created_at", "updated_at"}
}

// 软删除
func (u *User) Deleted() bool {
    return !u.DeletedAt.IsZero()
}
```

### 7.2 使用示例

```go
package service

type UserService struct {
    db *model.Model
}

func NewUserService() *UserService {
    return &UserService{
        db: model.New("users"),
    }
}

// CreateUser 创建用户
func (s *UserService) CreateUser(username, email, password string) (*model.User, error) {
    user := &model.User{
        Username: username,
        Email:    email,
        Password: password,  // 会自动加密
        Status:   1,
        Options: model.JSONField{
            "theme": "light",
            "lang":  "zh-CN",
        },
    }
    
    err := s.db.Insert(user)
    return user, err
}

// GetUserByID 获取用户
func (s *UserService) GetUserByID(id int) (*model.User, error) {
    user := &model.User{}
    err := s.db.Where("id", id).First(user)
    return user, err
}

// GetUserByEmail 通过邮箱获取用户
func (s *UserService) GetUserByEmail(email string) (*model.User, error) {
    user := &model.User{}
    err := s.db.WhereJSON("options", "lang", "zh-CN").
        Where("email", email).
        First(user)
    return user, err
}

// UpdateUser 更新用户
func (s *UserService) UpdateUser(user *model.User) error {
    // 只更新修改的字段
    return s.db.Where("id", user.ID).Update(user)
}

// DeleteUser 软删除用户
func (s *UserService) DeleteUser(id int) error {
    return s.db.Where("id", id).Delete()
}

// RestoreUser 恢复用户
func (s *UserService) RestoreUser(id int) error {
    return s.db.Where("id", id).RestoreByID(id)
}

// GetDeletedUsers 获取已删除用户
func (s *UserService) GetDeletedUsers() ([]*model.User, error) {
    var users []*model.User
    err := s.db.OnlyTrashed().Select(&users)
    return users, err
}
```

## 八、最佳实践

### 8.1 模型设计

```go
// 推荐：使用基础模型嵌入
type Article struct {
    model.Model
    ID      int    `json:"id"`
    Title   string `json:"title"`
}

// 不推荐：重复实现基础功能
type Article struct {
    ID        int
    Title     string
    CreatedAt time.Time
    UpdatedAt time.Time
    // ... 手动实现所有方法
}
```

---

### 8.2 JSON 字段使用

```go
// 推荐：定义类型别名
type UserOptions map[string]interface{}

type User struct {
    Options UserOptions `json:"options"`
}

// 使用
user.Options = UserOptions{
    "theme": "dark",
    "notifications": true,
}
```

---

### 8.3 软删除注意事项

```go
// 查询时注意是否包含已删除数据
// 默认不包含
articles := model.New("articles").Select()

// 明确指定
articles.WithTrashed().Select()      // 包含已删除
articles.WithoutTrashed().Select()   // 不包含已删除
articles.OnlyTrashed().Select()      // 只查询已删除
```

## 总结

Vigo ORM 增强提供：

✅ **软删除**: Delete/ForceDelete/Restore/Trashed  
✅ **JSON 查询**: WhereJSON/WhereJSONIn/WhereJSONLike  
✅ **自动时间戳**: CreatedAt/UpdatedAt 自动管理  
✅ **获取器修改器**: GetXAttribute/SetXAttribute  
✅ **只读字段**: Readonly 方法  
✅ **脏数据检测**: IsDirty/GetDirty  

合理使用 ORM 增强可以：
- 简化数据操作
- 提升代码质量
- 保证数据一致性
- 提高开发效率
