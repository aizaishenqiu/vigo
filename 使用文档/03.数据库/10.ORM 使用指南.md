# ORM 使用指南

Vigo 框架提供简洁高效的 ORM 功能，支持链式调用、结构体映射、事务处理等特性。

## 一、快速开始

### 1.1 定义模型

```go
package model

import "time"

// User 用户模型
type User struct {
    ID        int       `db:"id"`         // 主键
    Name      string    `db:"name"`       // 用户名
    Email     string    `db:"email"`      // 邮箱
    Status    string    `db:"status"`     // 状态
    CreatedAt time.Time `db:"created_at"` // 创建时间
    UpdatedAt time.Time `db:"updated_at"` // 更新时间
}

// TableName 指定表名（可选，默认使用结构体名复数形式）
func (u *User) TableName() string {
    return "users"
}
```

**数据库表结构**:
```sql
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    status VARCHAR(50) DEFAULT 'active',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
```

### 1.2 创建 ORM 实例

```go
import "vigo/framework/db"

// 创建 ORM 实例
orm := db.NewORM(db.GetDB())
```

## 二、查询操作

### 2.1 查询单条记录

```go
var user User
err := orm.Model(&User{}).Where("id", 1).First(&user)
if err != nil {
    if err == sql.ErrNoRows {
        fmt.Println("用户不存在")
    } else {
        log.Printf("查询失败: %v", err)
    }
}
fmt.Printf("用户: %+v\n", user)
```

### 2.2 查询多条记录

```go
var users []User
err := orm.Model(&User{}).Where("status", "active").Find(&users)
if err != nil {
    log.Printf("查询失败: %v", err)
}
for _, user := range users {
    fmt.Printf("用户: %+v\n", user)
}
```

### 2.3 条件查询

```go
// 等于条件
var users []User
err := orm.Model(&User{}).
    Where("status", "active").
    Find(&users)

// 使用操作符
err = orm.Model(&User{}).
    WhereOp("age", ">=", 18).
    WhereOp("age", "<=", 60).
    Find(&users)

// IN 条件
ids := []interface{}{1, 2, 3}
err = orm.Model(&User{}).
    WhereIn("id", ids).
    Find(&users)
```

### 2.4 排序和分页

```go
// 排序
var users []User
err := orm.Model(&User{}).
    Order("created_at DESC").
    Find(&users)

// 限制数量
err = orm.Model(&User{}).
    Order("id DESC").
    Limit(10).
    Find(&users)

// 分页
page := 1
pageSize := 15
err = orm.Model(&User{}).
    Where("status", "active").
    Order("id DESC").
    Page(page, pageSize).
    Find(&users)
```

### 2.5 统计数量

```go
count, err := orm.Model(&User{}).Where("status", "active").Count()
if err != nil {
    log.Printf("统计失败: %v", err)
}
fmt.Printf("活跃用户数量: %d\n", count)
```

## 三、写入操作

### 3.1 创建记录

```go
// 创建单个记录
user := User{
    Name:   "张三",
    Email:  "zhangsan@example.com",
    Status: "active",
}
err := orm.Model(&User{}).Create(&user)
if err != nil {
    log.Printf("创建失败: %v", err)
}
```

### 3.2 更新记录

```go
// 根据条件更新
err := orm.Model(&User{}).
    Where("id", 1).
    Update(map[string]interface{}{
        "name":   "李四",
        "status": "inactive",
    })
if err != nil {
    log.Printf("更新失败: %v", err)
}
```

### 3.3 删除记录

```go
// 根据条件删除（必须有 WHERE 条件）
err := orm.Model(&User{}).Where("id", 1).Delete()
if err != nil {
    log.Printf("删除失败: %v", err)
}

// 批量删除
err = orm.Model(&User{}).
    Where("status", "inactive").
    Delete()
```

## 四、事务处理

### 4.1 基本事务

```go
err := orm.Transaction(func(tx *db.ORM) error {
    // 创建用户
    user := User{Name: "张三", Email: "zhangsan@example.com"}
    if err := tx.Model(&User{}).Create(&user); err != nil {
        return err // 回滚事务
    }
    
    // 创建关联记录
    profile := Profile{UserID: user.ID, Bio: "简介"}
    if err := tx.Model(&Profile{}).Create(&profile); err != nil {
        return err // 回滚事务
    }
    
    return nil // 提交事务
})

if err != nil {
    log.Printf("事务失败: %v", err)
}
```

### 4.2 事务中的查询和更新

```go
err := orm.Transaction(func(tx *db.ORM) error {
    // 查询用户
    var user User
    if err := tx.Model(&User{}).Where("id", 1).First(&user); err != nil {
        return err
    }
    
    // 更新用户状态
    if err := tx.Model(&User{}).
        Where("id", user.ID).
        Update(map[string]interface{}{"status": "vip"}); err != nil {
        return err
    }
    
    // 插入日志
    log := Log{UserID: user.ID, Action: "upgrade_to_vip"}
    if err := tx.Model(&Log{}).Create(&log); err != nil {
        return err
    }
    
    return nil
})
```

## 五、结构体标签

### 5.1 标签说明

```go
type User struct {
    ID        int       `db:"id"`              // 映射到 id 字段
    Name      string    `db:"name"`            // 映射到 name 字段
    Email     string    `db:"email"`           // 映射到 email 字段
    Password  string    `db:"-"`               // - 表示忽略该字段
    CreatedAt time.Time `db:"created_at"`      // 映射到 created_at 字段
    UpdatedAt time.Time `db:"updated_at;omitempty"` // omitempty 表示零值时忽略
}
```

### 5.2 标签选项

- `db:"column_name"` - 指定数据库列名
- `db:"-"` - 忽略该字段
- `db:"column_name;omitempty"` - 零值时忽略（用于创建和更新）

## 六、高级用法

### 6.1 自定义表名

```go
// 方式1：实现 TableName 方法
type User struct {
    ID   int    `db:"id"`
    Name string `db:"name"`
}

func (u *User) TableName() string {
    return "sys_users" // 自定义表名
}

// 方式2：使用 Table 方法
orm.Table("custom_users").Where("id", 1).First(&user)
```

### 6.2 链式调用

```go
// 复杂的链式查询
var users []User
err := orm.Model(&User{}).
    Where("status", "active").
    WhereOp("age", ">=", 18).
    WhereIn("role", []interface{}{"admin", "user"}).
    Order("created_at DESC").
    Page(1, 20).
    Find(&users)
```

### 6.3 类型转换

ORM 自动处理以下类型转换：

```go
type User struct {
    ID        int       `db:"id"`          // int, int64 自动转换
    Name      string    `db:"name"`        // string
    Age       int       `db:"age"`         // 从 string/[]uint8 转换
    Score     float64   `db:"score"`       // float64
    IsActive  bool      `db:"is_active"`   // bool
    CreatedAt time.Time `db:"created_at"`  // time.Time
}
```

## 七、完整示例

### 7.1 用户管理服务

```go
package service

import (
    "database/sql"
    "vigo/framework/db"
)

type UserService struct {
    orm *db.ORM
}

func NewUserService() *UserService {
    return &UserService{
        orm: db.NewORM(db.GetDB()),
    }
}

// GetUserByID 根据ID获取用户
func (s *UserService) GetUserByID(id int) (*User, error) {
    var user User
    err := s.orm.Model(&User{}).Where("id", id).First(&user)
    if err != nil {
        return nil, err
    }
    return &user, nil
}

// GetActiveUsers 获取活跃用户列表
func (s *UserService) GetActiveUsers(page, pageSize int) ([]User, error) {
    var users []User
    err := s.orm.Model(&User{}).
        Where("status", "active").
        Order("created_at DESC").
        Page(page, pageSize).
        Find(&users)
    return users, err
}

// CreateUser 创建用户
func (s *UserService) CreateUser(name, email string) (*User, error) {
    user := User{
        Name:   name,
        Email:  email,
        Status: "active",
    }
    if err := s.orm.Model(&User{}).Create(&user); err != nil {
        return nil, err
    }
    return &user, nil
}

// UpdateUserStatus 更新用户状态
func (s *UserService) UpdateUserStatus(id int, status string) error {
    return s.orm.Model(&User{}).
        Where("id", id).
        Update(map[string]interface{}{"status": status})
}

// DeleteUser 删除用户
func (s *UserService) DeleteUser(id int) error {
    return s.orm.Model(&User{}).Where("id", id).Delete()
}

// TransferUser 转移用户（事务示例）
func (s *UserService) TransferUser(userID int, newDepartment string) error {
    return s.orm.Transaction(func(tx *db.ORM) error {
        // 更新用户部门
        if err := tx.Model(&User{}).
            Where("id", userID).
            Update(map[string]interface{}{"department": newDepartment}); err != nil {
            return err
        }
        
        // 记录转移日志
        log := TransferLog{
            UserID:      userID,
            NewDepartment: newDepartment,
        }
        if err := tx.Model(&TransferLog{}).Create(&log); err != nil {
            return err
        }
        
        return nil
    })
}
```

## 八、注意事项

1. **结构体字段必须导出**（首字母大写）才能被 ORM 识别
2. **删除操作必须有 WHERE 条件**，否则会报错
3. **事务中的错误要及时返回**，否则事务不会回滚
4. **使用指针传递**给 `First` 和 `Find` 方法
5. **表名默认使用结构体名的小写复数形式**（如 `User` -> `users`）

## 九、与查询构造器对比

| 特性 | ORM | 查询构造器 |
|------|-----|-----------|
| 结构体映射 | ✅ 自动 | ❌ 手动 |
| 链式调用 | ✅ 支持 | ✅ 支持 |
| 类型安全 | ✅ 是 | ❌ 否 |
| 代码量 | 少 | 多 |
| 灵活性 | 中等 | 高 |
| 性能 | 略低 | 略高 |

**建议**：
- 简单的 CRUD 操作使用 ORM
- 复杂的查询使用查询构造器
- 性能敏感的场景使用查询构造器
