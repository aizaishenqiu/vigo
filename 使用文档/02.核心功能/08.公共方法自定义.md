# 公共方法自定义详解

Vigo 框架允许用户在 `app/common` 目录中自定义公共方法，这些方法可以在全局任意位置调用，类似于 ThinkPHP 的 `common.php` 和 Laravel 的全局辅助函数。

---

## 一、什么是公共方法

### 1.1 公共方法概念

公共方法是用户自定义的全局工具函数，用于：

- **业务封装**: 将项目特有的业务逻辑封装为公共方法
- **代码复用**: 避免重复代码，提高复用性
- **统一规范**: 统一项目中常用的操作方式
- **全局调用**: 在控制器、模型、中间件等任意位置调用

### 1.2 与助手函数的区别

| 特性     | 框架助手函数        | 用户公共方法           |
| :------- | :------------------ | :--------------------- |
| **位置** | `framework/helper/` | `app/common/`          |
| **用途** | 通用工具函数        | 业务相关函数           |
| **修改** | 不建议修改          | 可自由定制             |
| **示例** | `helper.Md5()`      | `common.FormatPrice()` |

```go
// app/common/common.go
package common

import (
    "crypto/md5"
    "encoding/hex"
    "fmt"
    "vigo/framework/helper"
)

// FormatUsername 格式化用户名
// 将用户名转换为首字母大写的格式
func FormatUsername(username string) string {
    return helper.Ucfirst(helper.Trim(username))
}

// FormatPrice 格式化价格
// 将价格格式化为保留两位小数的字符串
func FormatPrice(price float64) string {
    return fmt.Sprintf("¥%.2f", price)
}

// GenerateOrderNo 生成订单号
// 格式：年月日时分秒 + 6 位随机数
func GenerateOrderNo() string {
    return helper.Datetime() + helper.RandomString(6)
}

// MaskPhone 手机号脱敏
// 将手机号中间 4 位替换为 *
func MaskPhone(phone string) string {
    if len(phone) != 11 {
        return phone
    }
    return phone[:3] + "****" + phone[7:]
}

// GeneratePassword 生成随机密码
func GeneratePassword(length int) string {
    return helper.RandomString(length)
}

// EncryptPassword 密码加密
func EncryptPassword(password string, salt string) string {
    hash := md5.Sum([]byte(password + salt))
    return hex.EncodeToString(hash[:])
}
```

```go
// app/common/string.go
package common

import (
    "regexp"
    "strings"
    "vigo/framework/helper"
)

// IsPhone 验证手机号
func IsPhone(phone string) bool {
    matched, _ := regexp.MatchString(`^1[3-9]\d{9}$`, phone)
    return matched
}

// IsEmail 验证邮箱
func IsEmail(email string) bool {
    matched, _ := regexp.MatchString(`^[\w-]+(\.[\w-]+)*@[\w-]+(\.[\w-]+)+$`, email)
    return matched
}

// IsIDCard 验证身份证号
func IsIDCard(idCard string) bool {
    matched, _ := regexp.MatchString(`^\d{17}[\dXx]$`, idCard)
    return matched
}

// HideEmail 隐藏邮箱
func HideEmail(email string) string {
    parts := strings.Split(email, "@")
    if len(parts) != 2 {
        return email
    }

    username := parts[0]
    domain := parts[1]

    if len(username) > 2 {
        username = username[:1] + strings.Repeat("*", len(username)-2) + username[len(username)-1:]
    }

    return username + "@" + domain
}

// CleanHTML 清理 HTML 标签
func CleanHTML(html string) string {
    return helper.StripTags(html)
}

// Slugify URL 友好化
func Slugify(str string) string {
    // 转小写
    str = strings.ToLower(str)
    // 替换空格为连字符
    str = strings.ReplaceAll(str, " ", "-")
    // 移除特殊字符
    reg := regexp.MustCompile(`[^\w-]+`)
    str = reg.ReplaceAllString(str, "")
    return str
}
```

```go
// app/common/number.go
package common

import (
    "math"
    "math/rand"
    "strconv"
    "time"
)

// Round 四舍五入
func Round(value float64, precision int) float64 {
    p := math.Pow10(precision)
    return math.Floor(value*p+0.5) / p
}

// FormatNumber 格式化数字（千分位）
func FormatNumber(value float64) string {
    intPart := int64(value)
    decPart := value - float64(intPart)

    str := strconv.FormatInt(intPart, 10)

    // 添加千分位
    result := ""
    for i, c := range str {
        if i > 0 && (len(str)-i)%3 == 0 {
            result += ","
        }
        result += string(c)
    }

    // 添加小数部分
    if decPart > 0 {
        result += fmt.Sprintf("%.2f", decPart)
    }

    return result
}

// RandomInt 生成随机整数
func RandomInt(min, max int) int {
    rand.Seed(time.Now().UnixNano())
    return rand.Intn(max-min+1) + min
}

// Percentage 计算百分比
func Percentage(part, total float64) float64 {
    if total == 0 {
        return 0
    }
    return (part / total) * 100
}

// Average 计算平均值
func Average(numbers []float64) float64 {
    if len(numbers) == 0 {
        return 0
    }

    sum := 0.0
    for _, num := range numbers {
        sum += num
    }

    return sum / float64(len(numbers))
}
```

```go
// app/common/file.go
package common

import (
    "io"
    "os"
    "path/filepath"
)

// FileExists 检查文件是否存在
func FileExists(path string) bool {
    _, err := os.Stat(path)
    return err == nil
}

// DirExists 检查目录是否存在
func DirExists(path string) bool {
    info, err := os.Stat(path)
    if err != nil {
        return false
    }
    return info.IsDir()
}

// CreateDir 创建目录
func CreateDir(path string) error {
    return os.MkdirAll(path, 0755)
}

// DeleteFile 删除文件
func DeleteFile(path string) error {
    return os.Remove(path)
}

// CopyFile 复制文件
func CopyFile(src, dst string) error {
    sourceFile, err := os.Open(src)
    if err != nil {
        return err
    }
    defer sourceFile.Close()

    destFile, err := os.Create(dst)
    if err != nil {
        return err
    }
    defer destFile.Close()

    _, err = io.Copy(destFile, sourceFile)
    return err
}

// GetFileSize 获取文件大小（字节）
func GetFileSize(path string) (int64, error) {
    info, err := os.Stat(path)
    if err != nil {
        return 0, err
    }
    return info.Size(), nil
}

// GetFileExtension 获取文件扩展名
func GetFileExtension(path string) string {
    return filepath.Ext(path)
}

// GetFileName 获取文件名（不含扩展名）
func GetFileName(path string) string {
    base := filepath.Base(path)
    ext := filepath.Ext(base)
    return base[:len(base)-len(ext)]
}
```

### 3.4 验证函数

```go
// app/common/validate.go
package common

import (
    "regexp"
    "strings"
    "unicode/utf8"
)

// ValidateUsername 验证用户名
func ValidateUsername(username string) (bool, string) {
    if len(username) < 3 {
        return false, "用户名至少 3 个字符"
    }

    if len(username) > 20 {
        return false, "用户名最多 20 个字符"
    }

    matched, _ := regexp.MatchString(`^[a-zA-Z][a-zA-Z0-9_]*$`, username)
    if !matched {
        return false, "用户名只能包含字母、数字和下划线，且必须以字母开头"
    }

    return true, ""
}

// ValidatePassword 验证密码
func ValidatePassword(password string) (bool, string) {
    if len(password) < 6 {
        return false, "密码至少 6 个字符"
    }

    if len(password) > 32 {
        return false, "密码最多 32 个字符"
    }

    // 检查是否包含字母和数字
    hasLetter := false
    hasDigit := false
    for _, c := range password {
        if (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') {
            hasLetter = true
        }
        if c >= '0' && c <= '9' {
            hasDigit = true
        }
    }

    if !hasLetter || !hasDigit {
        return false, "密码必须包含字母和数字"
    }

    return true, ""
}

// ValidateURL 验证 URL
func ValidateURL(url string) bool {
    matched, _ := regexp.MatchString(`^https?://[^\s]+$`, url)
    return matched
}

// ValidateIP 验证 IP 地址
func ValidateIP(ip string) bool {
    matched, _ := regexp.MatchString(`^(\d{1,3}\.){3}\d{1,3}$`, ip)
    if !matched {
        return false
    }

    // 验证每个段的范围
    parts := strings.Split(ip, ".")
    for _, part := range parts {
        if len(part) > 3 {
            return false
        }
        // 这里可以添加更严格的验证
    }

    return true
}

// IsEmpty 判断是否为空
func IsEmpty(value interface{}) bool {
    if value == nil {
        return true
    }

    switch v := value.(type) {
    case string:
        return strings.TrimSpace(v) == ""
    case []interface{}:
        return len(v) == 0
    case map[string]interface{}:
        return len(v) == 0
    default:
        return false
    }
}
```

## 四、使用公共方法

### 4.1 在控制器中使用

```go
// app/controller/user.go
package controller

import (
    "vigo/framework/mvc"
    "vigo/app/common"
    "vigo/app/model"
)

// Register 用户注册
func Register(c *mvc.Context) {
    username := c.PostForm("username")
    password := c.PostForm("password")
    phone := c.PostForm("phone")

    // 使用公共方法验证
    if valid, msg := common.ValidateUsername(username); !valid {
        c.Json(400, map[string]interface{}{
            "code":    400,
            "message": msg,
        })
        return
    }

    if valid, msg := common.ValidatePassword(password); !valid {
        c.Json(400, map[string]interface{}{
            "code":    400,
            "message": msg,
        })
        return
    }

    if !common.IsPhone(phone) {
        c.Json(400, map[string]interface{}{
            "code":    400,
            "message": "手机号格式错误",
        })
        return
    }

    // 使用公共方法加密密码
    passwordHash := common.EncryptPassword(password, "salt")

    // 创建用户
    user := &model.User{
        Username: common.FormatUsername(username),
        Password: passwordHash,
        Phone:    common.MaskPhone(phone),
    }

    // 生成邀请码
    user.InviteCode = common.GenerateInviteCode(user.ID)

    // 保存到数据库
    model.CreateUser(user)

    c.Json(200, map[string]interface{}{
        "code":    0,
        "message": "注册成功",
    })
}
```

### 4.3 在中间件中使用

```go
// app/middleware/auth.go
package middleware

import (
    "vigo/framework/mvc"
    "vigo/app/common"
    "vigo/framework/facade"
)

// Auth 认证中间件
func Auth() mvc.Middleware {
    return func(c *mvc.Context) {
        token := c.GetHeader("Authorization")

        if common.IsEmpty(token) {
            c.Json(401, map[string]interface{}{
                "code":    401,
                "message": "未授权",
            })
            c.Abort()
            return
        }

        // 验证 Token
        userID := facade.Redis.Get("token:" + token)
        if userID == "" {
            c.Json(401, map[string]interface{}{
                "code":    401,
                "message": "Token 已过期",
            })
            c.Abort()
            return
        }

        c.Set("user_id", userID)
        c.Next()
    }
}
```
